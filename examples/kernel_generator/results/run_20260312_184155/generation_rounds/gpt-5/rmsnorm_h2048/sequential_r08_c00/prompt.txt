// Read-only weight in constant memory: 2048 bfloat16 elements (4KB)
__constant__ uint16_t c_weight_bf16[HIDDEN_SIZE_2048];

// Fast bit-cast helpers for BF16 <-> FP32
static __device__ __forceinline__ float bf16_to_float(uint16_t h) {
  uint32_t u = static_cast<uint32_t>(h) << 16;
  return __uint_as_float(u);
}

static __device__ __forceinline__ uint16_t float_to_bf16_rne(float f) {
  uint32_t u = __float_as_uint(f);
  // Round to nearest even
  uint32_t lsb = (u >> 16) & 1u;
  u += 0x7FFFu + lsb;
  return static_cast<uint16_t>(u >> 16);
}

// Warp-level reduction: sum across 32 threads
static __device__ __forceinline__ float warp_sum(float v) {
  unsigned mask = 0xffffffffu;
  v += __shfl_down_sync(mask, v, 16);
  v += __shfl_down_sync(mask, v, 8);
  v += __shfl_down_sync(mask, v, 4);
  v += __shfl_down_sync(mask, v, 2);
  v += __shfl_down_sync(mask, v, 1);
  return v;
}

template<int BLOCK_SIZE, int HSIZE>
__launch_bounds__(BLOCK_SIZE, 2)
__global__ void rmsnorm_h2048_kernel(const uint16_t* __restrict__ x_bf16,
                                     const uint16_t* __restrict__ /*w_bf16_unused*/,
                                     uint16_t* __restrict__ y_bf16,
                                     int64_t batch_stride) {
  static_assert(HSIZE == 2048, "Kernel specialized for HSIZE=2048");
  static_assert((HSIZE % BLOCK_SIZE) == 0, "Hidden size must be divisible by block size");

  const int tid = threadIdx.x;
  const int wid = tid >> 5;   // warp id in block
  const int lane = tid & 31;  // lane id in warp
  const int num_warps = BLOCK_SIZE / 32;

  __shared__ float warp_sums[BLOCK_SIZE / 32];
  __shared__ float inv_rms_shared;

  const int64_t row = static_cast<int64_t>(blockIdx.x);
  const int64_t base = row * batch_stride;

  // Each thread processes ITER elements along the hidden dimension
  constexpr int ITER = HSIZE / BLOCK_SIZE;

  float x_cache[ITER];
  float partial_sum = 0.0f;

#pragma unroll
  for (int j = 0; j < ITER; ++j) {
    const int idx = j * BLOCK_SIZE + tid;
#if __CUDA_ARCH__ >= 700
    uint16_t hx = __ldg(&x_bf16[base + idx]);
#else
    uint16_t hx = x_bf16[base + idx];
#endif
    float xi = bf16_to_float(hx);
    x_cache[j] = xi;
    partial_sum = fmaf(xi, xi, partial_sum);
  }

  // Intra-warp reduction to compute sum of squares
  float warp_total = warp_sum(partial_sum);

  if (lane == 0) {
    warp_sums[wid] = warp_total;
  }
  __syncthreads();

  // Reduce warp totals in warp 0
  if (wid == 0) {
    float block_sum = (lane < num_warps) ? warp_sums[lane] : 0.0f;
    block_sum = warp_sum(block_sum);
    if (lane == 0) {
      constexpr float EPS = 1e-6f;
      // Precomputed reciprocal of HSIZE to avoid division latency
      constexpr float INV_HSIZE = 1.0f / static_cast<float>(HSIZE);
      float mean = block_sum * INV_HSIZE;
      inv_rms_shared = rsqrtf(mean + EPS);
    }
  }
  __syncthreads();

  const float inv_rms = inv_rms_shared;

  // Normalize and scale by weight (read from constant memory)
#pragma unroll
  for (int j = 0; j < ITER; ++j) {
    const int idx = j * BLOCK_SIZE + tid;
    // Constant memory read; indices differ across threads but data resides in L1/constant cache
    uint16_t hw = c_weight_bf16[idx];
    float wi = bf16_to_float(hw);
    float yi = (x_cache[j] * inv_rms) * wi;
    y_bf16[base + idx] = float_to_bf16_rne(yi);
  }
}

void rmsnorm_h2048_cuda(const uint16_t* hidden_states_bf16,
                        const uint16_t* weight_bf16,
                        uint16_t* output_bf16,
                        int64_t batch_size,
                        cudaStream_t stream) {
  if (batch_size <= 0) return;

  constexpr int H = HIDDEN_SIZE_2048;
  constexpr int BLOCK = 512;  // 4 elements/thread

  static_assert(H % BLOCK == 0, "Hidden size must be divisible by block size");

  // Copy weight to constant memory for read-only cached access by all blocks
