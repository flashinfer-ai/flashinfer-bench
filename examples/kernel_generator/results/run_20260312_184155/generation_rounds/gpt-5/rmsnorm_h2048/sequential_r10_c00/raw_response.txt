#ifndef RMSNORM_H2048_KERNEL_H_

#include <cuda_runtime.h>
#include <cstdint>

// Fixed constant for hidden size
constexpr int HIDDEN_SIZE_2048 = 2048;

// CUDA launcher: launches the RMSNorm kernel for hidden_size=2048.
// Operates on the provided device pointers without additional allocations.
// hidden_states_bf16: [batch_size, 2048] BF16 (as uint16_t storage)
// weight_bf16: [2048] BF16 (as uint16_t storage)
// output_bf16: [batch_size, 2048] BF16 (as uint16_t storage)
void rmsnorm_h2048_cuda(const uint16_t* hidden_states_bf16,
                        const uint16_t* weight_bf16,
                        uint16_t* output_bf16,
                        int64_t batch_size,
                        cudaStream_t stream);

// PyTorch binding entry point. Validates inputs, launches CUDA, returns output.
torch::Tensor run(torch::Tensor hidden_states, torch::Tensor weight);

#include "kernel.h"
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdio>

#ifndef CUDA_CHECK
#define CUDA_CHECK(stmt)                                                     \
  do {                                                                       \
    cudaError_t err__ = (stmt);                                              \
    if (err__ != cudaSuccess) {                                              \
      fprintf(stderr, "CUDA error %s at %s:%d: %s\n", #stmt, __FILE__, __LINE__, cudaGetErrorString(err__)); \
    }                                                                        \
  } while (0)
#endif

// Fast bit-cast helpers for BF16 <-> FP32
static __device__ __forceinline__ float bf16_to_float(uint16_t h) {
  uint32_t u = static_cast<uint32_t>(h) << 16;
  return __uint_as_float(u);
}

static __device__ __forceinline__ uint16_t float_to_bf16_rne(float f) {
  uint32_t u = __float_as_uint(f);
  // Round-to-nearest-even for bfloat16
  uint32_t lsb = (u >> 16) & 1u;
  u += 0x7FFFu + lsb;
  return static_cast<uint16_t>(u >> 16);
}

static __device__ __forceinline__ uint32_t pack_bf16x2(uint16_t lo, uint16_t hi) {
  return (static_cast<uint32_t>(hi) << 16) | static_cast<uint32_t>(lo);
}

// Warp-level reduction: sum across 32 threads
static __device__ __forceinline__ float warp_sum(float v) {
  unsigned mask = 0xffffffffu;
  v += __shfl_down_sync(mask, v, 16);
  v += __shfl_down_sync(mask, v, 8);
  v += __shfl_down_sync(mask, v, 4);
  v += __shfl_down_sync(mask, v, 2);
  v += __shfl_down_sync(mask, v, 1);
  return v;
}

template<int BLOCK_SIZE, int HSIZE>
__launch_bounds__(BLOCK_SIZE, 2)
__global__ void rmsnorm_h2048_kernel(const uint16_t* __restrict__ x_bf16,
                                     const uint16_t* __restrict__ w_bf16,
                                     uint16_t* __restrict__ y_bf16,
                                     int64_t batch_stride_elems) {
  static_assert(HSIZE == 2048, "Kernel specialized for HSIZE=2048");
  static_assert(BLOCK_SIZE == 1024, "This kernel expects BLOCK_SIZE == 1024 for 2 elems/thread");
  static_assert((HSIZE % (BLOCK_SIZE * 2)) == 0 || (HSIZE == BLOCK_SIZE * 2), "HSIZE must be 2 * BLOCK_SIZE");

  const int tid = threadIdx.x;
  const int wid = tid >> 5;   // warp id in block
  const int lane = tid & 31;  // lane id in warp
  const int num_warps = BLOCK_SIZE / 32;

  __shared__ float warp_sums[BLOCK_SIZE / 32];  // 32
  __shared__ float inv_rms_shared;

  const int64_t row = static_cast<int64_t>(blockIdx.x);
  const int64_t base_elem = row * batch_stride_elems;      // element index (bf16 elements)
  const int64_t base_u32  = base_elem >> 1;                // uint32 index (2 bf16 packed)

  const uint32_t* __restrict__ x_u32 = reinterpret_cast<const uint32_t*>(x_bf16);
  const uint32_t* __restrict__ w_u32 = reinterpret_cast<const uint32_t*>(w_bf16);
  uint32_t* __restrict__ y_u32 = reinterpret_cast<uint32_t*>(y_bf16);

  // Load two bfloat16 hidden states as one 32-bit transaction
#if __CUDA_ARCH__ >= 700
  uint32_t x_pack = __ldg(&x_u32[base_u32 + tid]);
#else
  uint32_t x_pack = x_u32[base_u32 + tid];
#endif

  uint16_t x0_bf16 = static_cast<uint16_t>(x_pack & 0xFFFFu);
  uint16_t x1_bf16 = static_cast<uint16_t>(x_pack >> 16);

  float x0 = bf16_to_float(x0_bf16);
  float x1 = bf16_to_float(x1_bf16);

  // Accumulate sum of squares for both elements in this thread
  float partial_sum = fmaf(x0, x0, 0.0f);
  partial_sum = fmaf(x1, x1, partial_sum);

  // Intra-warp reduction
  float warp_total = warp_sum(partial_sum);

  if (lane == 0) {
    warp_sums[wid] = warp_total;
  }
  __syncthreads();

  // Reduce warp totals in warp 0
  if (wid == 0) {
    float block_sum = (lane < num_warps) ? warp_sums[lane] : 0.0f;
    block_sum = warp_sum(block_sum);
    if (lane == 0) {
      constexpr float EPS = 1e-6f;
      constexpr float INV_HSIZE = 1.0f / static_cast<float>(HSIZE);
      float mean = block_sum * INV_HSIZE;
      inv_rms_shared = rsqrtf(mean + EPS);
    }
  }
  __syncthreads();

  const float inv_rms = inv_rms_shared;

  // Load two bfloat16 weights as one 32-bit transaction
#if __CUDA_ARCH__ >= 700
  uint32_t w_pack = __ldg(&w_u32[tid]);
#else
  uint32_t w_pack = w_u32[tid];
#endif

  uint16_t w0_bf16 = static_cast<uint16_t>(w_pack & 0xFFFFu);
  uint16_t w1_bf16 = static_cast<uint16_t>(w_pack >> 16);

  float w0 = bf16_to_float(w0_bf16);
  float w1 = bf16_to_float(w1_bf16);

  // Normalize and scale
  float y0 = (x0 * inv_rms) * w0;
  float y1 = (x1 * inv_rms) * w1;

  // Convert back to BF16 and store as packed uint32
  uint16_t y0_bf16 = float_to_bf16_rne(y0);
  uint16_t y1_bf16 = float_to_bf16_rne(y1);
  uint32_t y_pack = pack_bf16x2(y0_bf16, y1_bf16);

  y_u32[base_u32 + tid] = y_pack;
}

// CUDA host-side launcher
void rmsnorm_h2048_cuda(const uint16_t* hidden_states_bf16,
                        const uint16_t* weight_bf16,
                        uint16_t* output_bf16,
                        int64_t batch_size,
                        cudaStream_t stream) {
  if (batch_size <= 0) return;

  constexpr int H = HIDDEN_SIZE_2048;
  constexpr int BLOCK = 1024;  // 2 elements/thread with packed u32 loads/stores

  dim3 grid(static_cast<unsigned int>(batch_size));
  dim3 block(BLOCK);

  rmsnorm_h2048_kernel<BLOCK, H><<<grid, block, 0, stream>>>(
      hidden_states_bf16, weight_bf16, output_bf16, static_cast<int64_t>(H));

  CUDA_CHECK(cudaGetLastError());
}

<cpp_file name="main.cpp">
#include "kernel.h"

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/core/ScalarType.h>
#include <c10/core/TensorOptions.h>
#include <sstream>
#include <stdexcept>

static inline void check_common(const torch::Tensor& hidden_states,
                                const torch::Tensor& weight) {
  if (!hidden_states.defined() || !weight.defined()) {
    throw std::invalid_argument("rmsnorm_h2048: Inputs must be defined tensors");
  }
  if (hidden_states.dtype() != torch::kBFloat16) {
    throw std::invalid_argument("rmsnorm_h2048: hidden_states must be bfloat16 (torch.bfloat16)");
  }
  if (weight.dtype() != torch::kBFloat16) {
    throw std::invalid_argument("rmsnorm_h2048: weight must be bfloat16 (torch.bfloat16)");
  }
  if (hidden_states.dim() != 2) {
    throw std::invalid_argument("rmsnorm_h2048: hidden_states must be 2D [batch_size, hidden_size]");
  }
  if (weight.dim() != 1) {
    throw std::invalid_argument("rmsnorm_h2048: weight must be 1D [hidden_size]");
  }
  if (hidden_states.size(1) != HIDDEN_SIZE_2048) {
    std::ostringstream oss;
    oss << "rmsnorm_h2048: hidden_size must be " << HIDDEN_SIZE_2048
        << " but got " << hidden_states.size(1);
    throw std::invalid_argument(oss.str());
  }
  if (weight.size(0) != HIDDEN_SIZE_2048) {
    std::ostringstream oss;
    oss << "rmsnorm_h2048: weight size must be " << HIDDEN_SIZE_2048
        << " but got " << weight.size(0);
    throw std::invalid_argument(oss.str());
  }
  if (hidden_states.size(0) <= 0) {
    throw std::invalid_argument("rmsnorm_h2048: batch_size must be > 0");
  }
}

torch::Tensor run(torch::Tensor hidden_states, torch::Tensor weight) {
  // Validate shapes and dtypes
  check_common(hidden_states, weight);

  // Decide target CUDA device:
  // - If any input is on CUDA, use that device
  // - Otherwise, use current CUDA device
  torch::Device target_device = torch::Device(torch::kCUDA, at::cuda::current_device());
  if (hidden_states.device().is_cuda()) {
    target_device = hidden_states.device();
  } else if (weight.device().is_cuda()) {
    target_device = weight.device();
  }

  // Guard the target device
  at::cuda::CUDAGuard guard(target_device);

  // Move inputs to target CUDA device if needed
  torch::Tensor hs_cuda = hidden_states;
  torch::Tensor w_cuda = weight;
  bool moved_from_cpu = false;

  if (!hidden_states.device().is_cuda() || hidden_states.device() != target_device) {
    hs_cuda = hidden_states.to(target_device, /*non_blocking=*/false, /*copy=*/true);
    moved_from_cpu = moved_from_cpu || !hidden_states.device().is_cuda();
  }
  if (!weight.device().is_cuda() || weight.device() != target_device) {
    w_cuda = weight.to(target_device, /*non_blocking=*/false, /*copy=*/true);
    moved_from_cpu = moved_from_cpu || !weight.device().is_cuda();
  }

  // Ensure contiguous for coalesced memory access
  if (!hs_cuda.is_contiguous()) hs_cuda = hs_cuda.contiguous();
  if (!w_cuda.is_contiguous()) w_cuda = w_cuda.contiguous();

  const int64_t batch_size = hs_cuda.size(0);

  // Allocate output tensor on CUDA
  torch::Tensor output_cuda = torch::empty_like(hs_cuda);

  // Raw pointers (BF16 storage is 16-bit)
  const uint16_t* x_ptr = reinterpret_cast<const uint16_t*>(hs_cuda.data_ptr<c10::BFloat16>());
  const uint16_t* w_ptr = reinterpret_cast<const uint16_t*>(w_cuda.data_ptr<c10::BFloat16>());
  uint16_t* y_ptr = reinterpret_cast<uint16_t*>(output_cuda.data_ptr<c10::BFloat16>());

  // Launch CUDA kernel on the current stream
  cudaStream_t stream = at::cuda::getCurrentCUDAStream();
  rmsnorm_h2048_cuda(x_ptr, w_ptr, y_ptr, batch_size, stream);

  // Check for launch errors
  cudaError_t err = cudaGetLastError();
  if (err != cudaSuccess) {
    std::ostringstream oss;
    oss << "rmsnorm_h2048: CUDA kernel launch failed: " << cudaGetErrorString(err);
    throw std::runtime_error(oss.str());
  }

  // If original inputs were on CPU, copy results back to CPU
  if (moved_from_cpu) {
    return output_cuda.to(hidden_states.device(), /*non_blocking=*/false, /*copy=*/true);
  }
  return output_cuda;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run, "Kernel execution function",
        py::arg("hidden_states"), py::arg("weight"));
}
