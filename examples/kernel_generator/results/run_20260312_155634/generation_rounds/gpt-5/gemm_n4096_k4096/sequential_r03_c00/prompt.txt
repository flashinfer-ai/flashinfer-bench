#include "kernel.h"
#include <cstdint>
#include <algorithm>

using namespace nvcuda;

extern "C" __global__ __launch_bounds__(128, 2)
void gemm_n4096_k4096_wmma_kernel(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    __half* __restrict__ C,
    int M)
{
  // Block tile sizes
  constexpr int TILE_M = 128;   // block tile M
  constexpr int TILE_N = 128;   // block tile N
  constexpr int TILE_K = 16;    // wmma k dimension
  constexpr int WARP_TILE_M = 64;
  constexpr int WARP_TILE_N = 64;

  // Compute block origin
  const int blockRow = blockIdx.y * TILE_M;
  const int blockCol = blockIdx.x * TILE_N;

  // Thread and warp info
  const int warpId  = threadIdx.x / 32; // 0..3
  const int laneId  = threadIdx.x % 32;

  // We use 4 warps per block arranged as 2x2 to cover 128x128
  const int warpRow = warpId / 2; // 0..1
  const int warpCol = warpId % 2; // 0..1

  // Origin of this warp's 64x64 tile
  const int m_tile = blockRow + warpRow * WARP_TILE_M;
  const int n_tile = blockCol + warpCol * WARP_TILE_N;

  // Accumulator fragments: 4x4 to cover 64x64 per warp
  wmma::fragment<wmma::accumulator, 16, 16, 16, float> c_frags[4][4];

  // Initialize accumulators
  for (int i = 0; i < 4; ++i) {
    for (int j = 0; j < 4; ++j) {
      wmma::fill_fragment(c_frags[i][j], 0.0f);
    }
  }

  // Small per-warp shared scratch (16x16) for safe, masked stores
  __shared__ float warp_scratch[4][16 * 16];

  // Main K loop
  for (int k0 = 0; k0 < GEMM_K_CONST; k0 += TILE_K) {
    // Load A fragments for the 4 row segments if within bounds
    wmma::fragment<wmma::matrix_a, 16, 16, 16, __half, wmma::row_major> a_frags[4];
    bool a_valid[4];
    for (int i = 0; i < 4; ++i) {
      const int rowA = m_tile + i * 16;
      a_valid[i] = (rowA < M);
      if (a_valid[i]) {
        const __half* Aptr = A + rowA * GEMM_K_CONST + k0; // row-major ld=K
        wmma::load_matrix_sync(a_frags[i], Aptr, GEMM_K_CONST);
      }
    }

    // Load B fragments for the 4 col segments (B used as col-major KxN)
    wmma::fragment<wmma::matrix_b, 16, 16, 16, __half, wmma::col_major> b_frags[4];
    for (int j = 0; j < 4; ++j) {
      const int colB = n_tile + j * 16; // column in N
      const __half* Bptr = B + colB * GEMM_K_CONST + k0; // col-major: row=k0, col=colB, ld=K
      wmma::load_matrix_sync(b_frags[j], Bptr, GEMM_K_CONST);
    }

    // MMA accumulate only for valid A rows
    for (int i = 0; i < 4; ++i) {
      if (!a_valid[i]) continue;
      for (int j = 0; j < 4; ++j) {
        wmma::mma_sync(c_frags[i][j], a_frags[i], b_frags[j], c_frags[i][j]);
      }
    }
  }

  // Epilogue: store accumulators with masking using per-warp shared scratch
  for (int i = 0; i < 4; ++i) {
    for (int j = 0; j < 4; ++j) {
      // Store 16x16 fragment to per-warp scratch
      wmma::store_matrix_sync(&warp_scratch[warpId][0], c_frags[i][j], 16, wmma::mem_row_major);

      // Each thread writes a subset of the 256 elements with row bounds check
      for (int e = laneId; e < 256; e += 32) {
        const int r = e / 16;
        const int c = e % 16;

        const int globalRow = m_tile + i * 16 + r;
        const int globalCol = n_tile + j * 16 + c;

        if (globalRow < M) {
          const float val = warp_scratch[warpId][e];
          C[globalRow * GEMM_N_CONST + globalCol] = __float2half_rn(val);
        }
      }
      // Warp-synchronous programming; no __syncthreads() required since each warp
      // uses a disjoint scratch region indexed by warpId.
    }
  }
}

void launch_gemm_n4096_k4096(const at::Half* A,
                             const at::Half* B,
                             at::Half* C,
                             int M,
                             cudaStream_t stream)
{
  // Cast to __half device type
  const __half* dA = reinterpret_cast<const __half*>(A);
  const __half* dB = reinterpret_cast<const __half*>(B);
  __half* dC = reinterpret_cast<__half*>(C);

  // Configure launch: block 128 threads (4 warps), grid x fixed, grid y by M
  dim3 block(128, 1, 1);
  dim3 grid((GEMM_N_CONST + 128 - 1) / 128, (M + 128 - 1) / 128, 1); // x=32 for N=4096

  gemm_n4096_k4096_wmma_kernel<<<grid, block, 0, stream>>>(dA, dB, dC, M);

  cudaError_t err = cudaGetLastError();
  if (err != cudaSuccess) {
    fprintf(stderr, "Kernel launch failed: %s\n", cudaGetErrorString(err));
  }
