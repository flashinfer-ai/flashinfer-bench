#include "kernel.h"
#include <cstdint>

using namespace nvcuda;

////////////////////////////////////////////////////////////////////////////////
// WMMA-based GEMM kernel with shared-memory staging.
// Computes: C[M,4096] = A[M,4096] * B[4096,4096]^T
// Inputs:
//  - A: row-major [M, K=4096]
//  - B: row-major [N=4096, K=4096], we treat as (K x N) col-major without copy
// Output:
//  - C: row-major [M, N=4096]
////////////////////////////////////////////////////////////////////////////////

extern "C" __global__ __launch_bounds__(128, 2)
void gemm_n4096_k4096_wmma_kernel(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    __half* __restrict__ C,
    int M)
{
  // Block tile sizes
  constexpr int TILE_M = 128;    // block tile in M
  constexpr int TILE_N = 128;    // block tile in N
  constexpr int TILE_K = 32;     // stage along K per iteration (multiple of 16 for WMMA)
  constexpr int WARP_TILE_M = 64;
  constexpr int WARP_TILE_N = 64;

  // Compute block origin
  const int blockRow = blockIdx.y * TILE_M; // in M
  const int blockCol = blockIdx.x * TILE_N; // in N

  // Thread and warp info
  const int tid     = threadIdx.x;          // 0..127
  const int warpId  = tid >> 5;             // 0..3 (4 warps per block)
  const int laneId  = tid & 31;

  // Warp arrangement 2x2 per 128x128 tile
  const int warpRow = warpId >> 1;          // 0..1
  const int warpCol = warpId & 1;           // 0..1

  // Origin of this warp's 64x64 tile in C
  const int m_tile = blockRow + warpRow * WARP_TILE_M;
  const int n_tile = blockCol + warpCol * WARP_TILE_N;

  // Shared memory: A tile [128 x TILE_K] in row-major, B tile [TILE_K x 128] in col-major
  __shared__ __align__(16) __half smemA[TILE_M * TILE_K];  // ld = TILE_K
  __shared__ __align__(16) __half smemB[TILE_K * TILE_N];  // ld = TILE_K (col-major view)

  // Accumulator fragments: 4x4 sub-tiles to cover 64x64 per warp
  wmma::fragment<wmma::accumulator, 16, 16, 16, float> c_frags[4][4];
  #pragma unroll
  for (int i = 0; i < 4; ++i) {
    #pragma unroll
    for (int j = 0; j < 4; ++j) {
      wmma::fill_fragment(c_frags[i][j], 0.0f);
    }
  }

  // Per-warp scratch for safe, masked epilogue stores (16x16 float tile)
  __shared__ float warp_scratch[4][16 * 16];

  // Loop over K dimension in chunks of TILE_K (32)
  for (int k0 = 0; k0 < GEMM_K_CONST; k0 += TILE_K) {

    // Cooperative load A tile: [TILE_M x TILE_K] from global to smemA (row-major)
    // Each thread loads multiple elements
    for (int idx = tid; idx < TILE_M * TILE_K; idx += blockDim.x) {
      int r = idx / TILE_K;        // 0..127
      int c = idx % TILE_K;        // 0..31
      int gRow = blockRow + r;
      int gCol = k0 + c;           // along K
      __half val = __float2half(0.0f);
      if (gRow < M) {
        // Row-major A: ld = GEMM_K_CONST
        val = A[gRow * GEMM_K_CONST + gCol];
      }
      smemA[r * TILE_K + c] = val;
    }

    // Cooperative load B tile: [TILE_K x TILE_N] from global to smemB (col-major)
    // We need B(n, k) -> B_col(k, n) where B is row-major [N x K]
    for (int idx = tid; idx < TILE_K * TILE_N; idx += blockDim.x) {
      int r = idx % TILE_K;        // 0..31 (k within this tile)
      int c = idx / TILE_K;        // 0..127 (n within this tile)
      int gN = blockCol + c;       // global N index
      int gK = k0 + r;             // global K index
      // Row-major B: ld = GEMM_K_CONST, element = B[gN, gK]
      __half val = B[gN * GEMM_K_CONST + gK];
      // Store in smemB as col-major [TILE_K x TILE_N], ld = TILE_K
      smemB[r + c * TILE_K] = val;
    }

    __syncthreads();

    // For each TILE_K chunk (split into two WMMA-k=16 sub-steps)
    #pragma unroll
    for (int kk = 0; kk < TILE_K; kk += 16) {
      // Load A fragments for the 4 row segments of this warp (64 rows => 4x16)
      wmma::fragment<wmma::matrix_a, 16, 16, 16, __half, wmma::row_major> a_frags[4];
      #pragma unroll
      for (int i = 0; i < 4; ++i) {
        // Pointer to 16x16 block in smemA at rows (warpRow*64 + i*16), cols (kk..kk+15)
        const __half* a_ptr = &smemA[(warpRow * WARP_TILE_M + i * 16) * TILE_K + kk];
        wmma::load_matrix_sync(a_frags[i], a_ptr, TILE_K);
      }

      // Load B fragments for the 4 column segments of this warp (64 cols => 4x16)
      wmma::fragment<wmma::matrix_b, 16, 16, 16, __half, wmma::col_major> b_frags[4];
      #pragma unroll
      for (int j = 0; j < 4; ++j) {
        // Pointer to 16x16 block in smemB at rows (kk..kk+15), cols (warpCol*64 + j*16)
        const __half* b_ptr = &smemB[kk + (warpCol * WARP_TILE_N + j * 16) * TILE_K];
        wmma::load_matrix_sync(b_frags[j], b_ptr, TILE_K);
      }

      // MMA accumulate
      #pragma unroll
      for (int i = 0; i < 4; ++i) {
        // If the target rows are beyond M, the smemA rows are zero; mma will add zeros.
        #pragma unroll
        for (int j = 0; j < 4; ++j) {
          wmma::mma_sync(c_frags[i][j], a_frags[i], b_frags[j], c_frags[i][j]);
        }
      }
    }

    __syncthreads();
  } // end K loop

  // Epilogue: write accumulators to global memory C with row-bound checks
  #pragma unroll
  for (int i = 0; i < 4; ++i) {
    #pragma unroll
    for (int j = 0; j < 4; ++j) {
      // Store 16x16 fragment to per-warp shared scratch (row-major, ld=16)
      wmma::store_matrix_sync(&warp_scratch[warpId][0], c_frags[i][j], 16, wmma::mem_row_major);

      // Each lane writes a subset of the 256 elements with row bounds check
      for (int e = laneId; e < 256; e += 32) {
        const int r = e / 16;
        const int c = e % 16;

        const int globalRow = m_tile + i * 16 + r;
        const int globalCol = n_tile + j * 16 + c;

        if (globalRow < M) {
          float val = warp_scratch[warpId][e];
          C[globalRow * GEMM_N_CONST + globalCol] = __float2half_rn(val);
        }
      }
      // Warps use disjoint scratch regions; no inter-warp sync needed.
    }
  }
}

// Host launcher implementation
void launch_gemm_n4096_k4096(const at::Half* A,
                             const at::Half* B,
                             at::Half* C,
                             int M,
                             cudaStream_t stream)
{
  const __half* dA = reinterpret_cast<const __half*>(A);
  const __half* dB = reinterpret_cast<const __half*>(B);
  __half* dC = reinterpret_cast<__half*>(C);

  // Configure launch: 128 threads (4 warps), 128x128 tiles across N and M
  dim3 block(128, 1, 1);
  dim3 grid((GEMM_N_CONST + 128 - 1) / 128, (M + 128 - 1) / 128, 1);

  gemm_n4096_k4096_wmma_kernel<<<grid, block, 0, stream>>>(dA, dB, dC, M);

  cudaError_t err = cudaGetLastError();
  if (err != cudaSuccess) {
    fprintf(stderr, "Kernel launch failed: %s\n", cudaGetErrorString(err));
  }
}
