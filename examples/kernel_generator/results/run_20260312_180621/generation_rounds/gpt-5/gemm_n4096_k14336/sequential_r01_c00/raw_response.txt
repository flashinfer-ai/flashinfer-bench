#include "kernel.h"

using namespace nvcuda;

#ifndef CUDA_CHECK
#define CUDA_CHECK(ans) { gpuAssert((ans), __FILE__, __LINE__); }
inline void gpuAssert(cudaError_t code, const char* file, int line, bool abort=true)
{
    if (code != cudaSuccess) {
        fprintf(stderr, "CUDA Error: %s at %s:%d\n", cudaGetErrorString(code), file, line);
        if (abort) { exit(code); }
    }
}
#endif

// WMMA-tiled GEMM kernel specialized for:
// A [M x 14336] row-major (half), B [4096 x 14336] row-major (half), C [M x 4096] row-major (half)
// Computes: C = A * B^T
// N and K are compile-time constants to allow better optimization.
template<int WARPS_M, int WARPS_N>
__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         int64_t M)
{
    // WMMA tile sizes
    constexpr int WM = 16;
    constexpr int WN = 16;
    constexpr int WK = 16;
    constexpr int WARPS_PER_BLOCK = WARPS_M * WARPS_N;

    // Threadblock tile sizes
    constexpr int TILE_M = WARPS_M * WM; // 64
    constexpr int TILE_N = WARPS_N * WN; // 64
    constexpr int TILE_K = WK;           // 16

    // Shared memory buffers:
    // A tile: [TILE_M x TILE_K] in row-major
    // B tile: [TILE_K x TILE_N] in "col-major" for WMMA B fragment (we store as [Kstep][Nstep])
    __shared__ __half As[TILE_M][TILE_K];
    __shared__ __half Bs[TILE_K][TILE_N];
    // Per-warp staging buffer to store accumulator fragments (float) in row-major
    __shared__ float Csmem[WARPS_PER_BLOCK][WM * WN];

    const int block_m = blockIdx.y; // along M
    const int block_n = blockIdx.x; // along N (4096)

    const int tid    = threadIdx.x;
    const int warpId = tid >> 5;      // warp index within block
    const int laneId = tid & 31;      // lane index within warp

    const int warp_m = warpId / WARPS_N; // warp's row tile within TB
    const int warp_n = warpId % WARPS_N; // warp's col tile within TB

    // Initialize the accumulator fragment
    wmma::fragment<wmma::accumulator, WM, WN, WK, float> c_frag;
    wmma::fill_fragment(c_frag, 0.0f);

    // Loop over K dimension in steps of TILE_K (16)
    for (int k0 = 0; k0 < K_CONST; k0 += TILE_K) {
        // Load A tile [TILE_M x TILE_K]
        // Cooperative load by all threads in the block
        for (int i = tid; i < TILE_M * TILE_K; i += blockDim.x) {
            int row = i / TILE_K;
            int col = i % TILE_K;
            int64_t g_row = static_cast<int64_t>(block_m) * TILE_M + row;
            int64_t g_col = k0 + col;
            __half val = __float2half(0.0f);
            if (g_row < M) {
                // A is [M x K_CONST] row-major
                val = A[g_row * K_CONST + g_col];
            }
            As[row][col] = val;
        }

        // Load B tile [TILE_K x TILE_N], sourced from B [N_CONST x K_CONST] row-major
        // We arrange Bs as [Kstep][Nstep], which matches WMMA B col-major with ld = TILE_K
        for (int i = tid; i < TILE_K * TILE_N; i += blockDim.x) {
            int row = i / TILE_N; // 0..TILE_K-1 (k within tile)
            int col = i % TILE_N; // 0..TILE_N-1 (n within tile)
            int64_t g_k = k0 + row;
            int64_t g_n = static_cast<int64_t>(block_n) * TILE_N + col;
            // B is [N_CONST x K_CONST] row-major, so element (n,k) at n*K + k
            Bs[row][col] = B[g_n * K_CONST + g_k];
        }

        __syncthreads();

        // Each warp computes a [WM x WN] tile of C
        // Load A and B fragments from shared memory and perform MMA
        {
            const __half* a_tile_ptr = &As[warp_m * WM][0];
            const __half* b_tile_ptr = &Bs[0][warp_n * WN];

            wmma::fragment<wmma::matrix_a, WM, WN, WK, __half, wmma::row_major> a_frag;
            wmma::fragment<wmma::matrix_b, WM, WN, WK, __half, wmma::col_major> b_frag;

            // Leading dimensions: A ld = TILE_K, B ld = TILE_K (for col_major)
            wmma::load_matrix_sync(a_frag, a_tile_ptr, TILE_K);
            wmma::load_matrix_sync(b_frag, b_tile_ptr, TILE_K);

            wmma::mma_sync(c_frag, a_frag, b_frag, c_frag);
        }

        __syncthreads(); // Ensure all warps are done with shared memory before next load
    }

    // Store accumulator fragment to per-warp shared memory (row-major WM x WN)
    wmma::store_matrix_sync(&Csmem[warpId][0], c_frag, WN, wmma::mem_row_major);

    // Each warp writes its 16x16 tile from shared memory to global memory, converting to half
    const int64_t base_m = static_cast<int64_t>(block_m) * TILE_M + warp_m * WM;
    const int64_t base_n = static_cast<int64_t>(block_n) * TILE_N + warp_n * WN;

    for (int i = laneId; i < WM * WN; i += 32) {
        int r = i / WN;
        int c = i % WN;
        int64_t gm = base_m + r;
        int64_t gn = base_n + c;
        if (gm < M) {
            float v = Csmem[warpId][r * WN + c];
            C[gm * N_CONST + gn] = __float2half_rn(v);
        }
    }
}

// Launcher implementation
void launch_gemm_n4096_k14336(const __half* A,
                              const __half* B,
                              __half* C,
                              int64_t M,
                              cudaStream_t stream)
{
    // Configure a 64x64 tile per block with 16 warps (4x4)
    constexpr int WARPS_M = 4;
    constexpr int WARPS_N = 4;
    constexpr int WARPS_PER_BLOCK = WARPS_M * WARPS_N;
    constexpr int THREADS_PER_BLOCK = WARPS_PER_BLOCK * 32;
    constexpr int TILE_M = WARPS_M * 16; // 64
    constexpr int TILE_N = WARPS_N * 16; // 64

    int grid_x = N_CONST / TILE_N; // 4096 / 64 = 64
    int grid_y = static_cast<int>((M + TILE_M - 1) / TILE_M);

    dim3 grid(grid_x, grid_y, 1);
    dim3 block(THREADS_PER_BLOCK, 1, 1);

    // Launch
    gemm_n4096_k14336_kernel<WARPS_M, WARPS_N><<<grid, block, 0, stream>>>(A, B, C, M);

    // Error check (asynchronous)
    CUDA_CHECK(cudaGetLastError());
}

#include "kernel.h"

#include &lt;torch/extension.h&gt;
#include &lt;ATen/cuda/CUDAContext.h&gt;
#include &lt;cuda.h&gt;
#include &lt;cuda_runtime.h&gt;
#include &lt;stdexcept&gt;
#include &lt;sstream&gt;

#ifndef CUDA_CHECK
#define CUDA_CHECK(ans) { gpuAssert_host((ans), __FILE__, __LINE__); }
inline void gpuAssert_host(cudaError_t code, const char* file, int line, bool abort=true)
{
    if (code != cudaSuccess) {
        std::ostringstream oss;
        oss &lt;&lt; "CUDA Error: " &lt;&lt; cudaGetErrorString(code) &lt;&lt; " at " &lt;&lt; file &lt;&lt; ":" &lt;&lt; line;
        if (abort) {
            throw std::runtime_error(oss.str());
        } else {
            fprintf(stderr, "%s\n", oss.str().c_str());
        }
    }
}
#endif

static inline void validate_inputs(const torch::Tensor&amp; A, const torch::Tensor&amp; B) {
    // Device checks
    if (!A.device().is_cuda() || !B.device().is_cuda()) {
        throw std::invalid_argument("Inputs A and B must be CUDA tensors.");
    }
    // DType checks
    if (A.dtype() != torch::kHalf) {
        throw std::invalid_argument("Input A must be of dtype torch.float16 (Half).");
    }
    if (B.dtype() != torch::kHalf) {
        throw std::invalid_argument("Input B must be of dtype torch.float16 (Half).");
    }
    // Dim checks
    if (A.dim() != 2 || B.dim() != 2) {
        throw std::invalid_argument("Inputs A and B must be 2D tensors.");
    }
    // Shape checks
    if (A.size(1) != K_CONST) {
        std::ostringstream oss;
        oss &lt;&lt; "Input A second dimension must be " &lt;&lt; K_CONST &lt;&lt; ", got " &lt;&lt; A.size(1) &lt;&lt; ".";
        throw std::invalid_argument(oss.str());
    }
    if (B.size(0) != N_CONST || B.size(1) != K_CONST) {
        std::ostringstream oss;
        oss &lt;&lt; "Input B shape must be [" &lt;&lt; N_CONST &lt;&lt; ", " &lt;&lt; K_CONST &lt;&lt; "], got [" &lt;&lt; B.size(0) &lt;&lt; ", " &lt;&lt; B.size(1) &lt;&lt; "].";
        throw std::invalid_argument(oss.str());
    }
    // Ensure same device
    if (A.get_device() != B.get_device()) {
        throw std::invalid_argument("Inputs A and B must be on the same CUDA device.");
    }
}

torch::Tensor run(torch::Tensor A, torch::Tensor B) {
    // Validate
    validate_inputs(A, B);

    // Make contiguous for expected row-major memory access
    if (!A.is_contiguous()) A = A.contiguous();
    if (!B.is_contiguous()) B = B.contiguous();

    const int64_t M = A.size(0);
    const auto device = A.device();

    // Allocate output tensor C [M x 4096] on same device, dtype half
    auto C = torch::empty({M, static_cast<int64_t>(N_CONST)}, A.options().dtype(torch::kHalf).device(device));

    // Get raw pointers as __half*
    const __half* A_ptr = reinterpret_cast<const __half*>(A.data_ptr&lt;at::Half&gt;());
    const __half* B_ptr = reinterpret_cast<const __half*>(B.data_ptr&lt;at::Half&gt;());
    __half* C_ptr       = reinterpret_cast<__half*>(C.data_ptr&lt;at::Half&gt;());

    // Current CUDA stream from PyTorch
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    // Launch kernel
    launch_gemm_n4096_k14336(A_ptr, B_ptr, C_ptr, M, stream);

    // Optionally synchronize errors on the current stream
    CUDA_CHECK(cudaGetLastError());
    // Do not force device-wide sync to keep async behavior with PyTorch unless needed
    // If strict sync is required, uncomment the next line:
    // CUDA_CHECK(cudaStreamSynchronize(stream));

    return C;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("run", &run, "Kernel execution function");
}
