#include "kernel.h"
#include <cstdio>
#include <cassert>

#ifndef CUDA_CHECK
#define CUDA_CHECK(err) do { \
  cudaError_t err_ = (err); \
  if (err_ != cudaSuccess) { \
    printf("CUDA Error %s at %s:%d\n", cudaGetErrorString(err_), __FILE__, __LINE__); \
  } \
} while (0)
#endif

// Tuned tiling parameters for A800 (Ampere)
static constexpr int BM = 128;   // Block tile size in M
static constexpr int BN = 128;   // Block tile size in N
static constexpr int BK = 128;   // Block tile size in K
static constexpr int TM = 8;     // Per-thread tile size in M
static constexpr int TN = 4;     // Per-thread tile size in N

// Padding to reduce shared memory bank conflicts (must be multiples of 8 for vectorized stores)
static constexpr int PAD_A = 8;  // pad shared K dimension for A tiles
static constexpr int PAD_B = 8;  // pad shared N dimension for B tiles

// Threadblock: 32 x 16 = 512 threads (16 warps)
static constexpr int BLOCK_THREADS_X = 32;
static constexpr int BLOCK_THREADS_Y = 16;

// Utility to get ceil-div
__host__ __device__ __forceinline__ int ceil_div_int(int a, int b) {
  return (a + b - 1) / b;
}

// Kernel implementation
__launch_bounds__(BLOCK_THREADS_X * BLOCK_THREADS_Y, 2)
__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         int M, int lda, int ldb, int ldc) {
  // Shared memory double-buffered storage
  // Layout:
  //  As buffers: 2 x (BM x (BK + PAD_A))
  //  Bs buffers: 2 x (BK x (BN + PAD_B))
  extern __shared__ __half shared_mem[];
  const int BKp = BK + PAD_A;
  const int BNp = BN + PAD_B;
  __half* As0 = shared_mem;
  __half* Bs0 = As0 + (BM * BKp);
  __half* As1 = Bs0 + (BK * BNp);
  __half* Bs1 = As1 + (BM * BKp);

  // Block coordinates
  const int block_n = blockIdx.x; // along N dimension
  const int block_m = blockIdx.y; // along M dimension

  const int m_block_origin = block_m * BM;
  const int n_block_origin = block_n * BN;

  // Thread identifiers
  const int lane = threadIdx.x;                         // 0..31
  const int warp_id = threadIdx.y;                      // 0..15 (each y is a warp)
  const int tid = warp_id * BLOCK_THREADS_X + lane;     // 0..511

  // Warp tile coordinates inside the block
  const int warp_m = warp_id / 4; // 0..3
  const int warp_n = warp_id % 4; // 0..3

  // Per-thread micro-tile offsets within the warp tile (32x32)
  const int row_group = (lane & 0x3);     // 0..3
  const int col_group = (lane >> 2);      // 0..7

  const int thread_row_offset = warp_m * 32 + row_group * TM; // warp_m*32 + 0,8,16,24
  const int thread_col_offset = warp_n * 32 + col_group * TN; // warp_n*32 + 0,4,8,...,28

  // Accumulator registers
  float acc[TM][TN];
  #pragma unroll
  for (int i = 0; i < TM; ++i) {
    #pragma unroll
    for (int j = 0; j < TN; ++j) {
      acc[i][j] = 0.0f;
    }
  }

  // Total tiles along K
  const int num_k_tiles = GEMM_K / BK; // 14336/128 = 112

  // Prepare leading dimensions
  const int lda_ = lda; // == GEMM_K
  const int ldb_ = ldb; // == GEMM_K
  const int ldc_ = ldc; // == GEMM_N

  // Helper lambda to load a tile of A and B into shared memory (vectorized by int4 = 8 half)
  auto load_tiles = [&](__half* As, __half* Bs, int k_tile) {
    // Global starting indices for this tile
    const int k0 = k_tile * BK;

    // Number of int4 (16B) loads per matrix per thread
    // Each int4 corresponds to 8 half elements
    const int total_A_int4 = (BM * BK) / 8; // 16384 / 8 = 2048
    const int total_B_int4 = (BK * BN) / 8; // 16384 / 8 = 2048
    const int loads_per_thread_A = total_A_int4 / (BLOCK_THREADS_X * BLOCK_THREADS_Y); // 2048 / 512 = 4
    const int loads_per_thread_B = total_B_int4 / (BLOCK_THREADS_X * BLOCK_THREADS_Y); // 2048 / 512 = 4

    // Load A tile: As[ row , col ], 0<=row<BM, 0<=col<BK, stored with stride BKp
    #pragma unroll
    for (int i = 0; i < loads_per_thread_A; ++i) {
      int idx4 = tid + i * (BLOCK_THREADS_X * BLOCK_THREADS_Y); // 0..2047
      int elem_idx = idx4 * 8; // half elements index into BM*BK tile
      int a_row = elem_idx / BK;      // 0..127
      int a_col = elem_idx % BK;      // 0..127

      int g_m = m_block_origin + a_row;
      int g_k = k0 + a_col;

      __half* s_ptr = As + a_row * BKp + a_col;
      if (g_m < M) {
        const __half* g_ptr_h = A + g_m * lda_ + g_k;
        // vectorized int4 load/store (16B, 8 half)
        const int4* g_ptr = reinterpret_cast<const int4*>(g_ptr_h);
        int4 val = *g_ptr;
        int4* s_ptr_v = reinterpret_cast<int4*>(s_ptr);
        *s_ptr_v = val;
      } else {
        // Zero-pad rows outside M
        int4 zero = make_int4(0, 0, 0, 0);
        int4* s_ptr_v = reinterpret_cast<int4*>(s_ptr);
        *s_ptr_v = zero;
      }
    }

    // Load B tile: Bs[ row(k) , col(n) ], 0<=row<BK, 0<=col<BN, stored with stride BNp
    #pragma unroll
    for (int i = 0; i < loads_per_thread_B; ++i) {
      int idx4 = tid + i * (BLOCK_THREADS_X * BLOCK_THREADS_Y);
      int elem_idx = idx4 * 8; // half elements
      int b_row = elem_idx / BN;      // along K-dimension (0..127)
      int b_col = elem_idx % BN;      // along N-dimension (0..127)

      int g_k = k0 + b_row;
      int g_n = n_block_origin + b_col;

      const __half* g_ptr_h = B + g_n * ldb_ + g_k; // B[n, k]
      __half* s_ptr = Bs + b_row * BNp + b_col;

      // All B indices are in-bounds for given constraints
      const int4* g_ptr = reinterpret_cast<const int4*>(g_ptr_h);
      int4 val = *g_ptr;
      int4* s_ptr_v = reinterpret_cast<int4*>(s_ptr);
      *s_ptr_v = val;
    }
  };

  // Preload first tiles
  load_tiles(As0, Bs0, 0);
  __syncthreads();

  // Loop over K tiles
  for (int kt = 0; kt < num_k_tiles; ++kt) {
    __half* As_cur = (kt % 2 == 0) ? As0 : As1;
    __half* Bs_cur = (kt % 2 == 0) ? Bs0 : Bs1;

    // Compute on current tiles
    #pragma unroll 4
    for (int kk = 0; kk < BK; ++kk) {
      float a_frag[TM];
      float b_frag[TN];

      // Load A fragment: TM rows for this thread at kk
      #pragma unroll
      for (int i = 0; i < TM; ++i) {
        int smem_row = thread_row_offset + i;
        a_frag[i] = __half2float(As_cur[smem_row * BKp + kk]);
      }
      // Load B fragment: TN cols for this thread at kk
      #pragma unroll
      for (int j = 0; j < TN; ++j) {
        int smem_col = thread_col_offset + j;
        b_frag[j] = __half2float(Bs_cur[kk * BNp + smem_col]);
      }

      // FMA accumulate
      #pragma unroll
      for (int i = 0; i < TM; ++i) {
        float a_val = a_frag[i];
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
          acc[i][j] = fmaf(a_val, b_frag[j], acc[i][j]);
        }
      }
    }

    // Preload next tiles if any
    if (kt + 1 < num_k_tiles) {
      __syncthreads();
      if ((kt % 2) == 0) {
        load_tiles(As1, Bs1, kt + 1);
      } else {
        load_tiles(As0, Bs0, kt + 1);
      }
      __syncthreads();
    }
  }

  // Write back C
  // Each thread writes its TM x TN micro-tile
  #pragma unroll
  for (int i = 0; i < TM; ++i) {
    int g_m = m_block_origin + thread_row_offset + i;
    if (g_m < M) {
      __half* C_row = C + g_m * ldc_ + n_block_origin + thread_col_offset;
      #pragma unroll
      for (int j = 0; j < TN; ++j) {
        C_row[j] = __float2half(acc[i][j]);
      }
    }
  }
}

// Host launcher
void gemm_n4096_k14336_cuda(const __half* A,
                            const __half* B,
                            __half* C,
                            int M,
                            int lda, int ldb, int ldc,
                            cudaStream_t stream) {
  // Grid and block configuration
  const int grid_x = GEMM_N / BN;               // 4096 / 128 = 32
  const int grid_y = ceil_div_int(M, BM);
  dim3 grid(grid_x, grid_y, 1);
  dim3 block(BLOCK_THREADS_X, BLOCK_THREADS_Y, 1);

  // Dynamic shared memory size
  const int BKp = BK + PAD_A;
  const int BNp = BN + PAD_B;
  size_t shared_bytes = sizeof(__half) * (2 * (BM * BKp) + 2 * (BK * BNp));

  // Set attributes for large shared memory per block on Ampere
  CUDA_CHECK(cudaFuncSetAttribute(
      gemm_n4096_k14336_kernel,
      cudaFuncAttributeMaxDynamicSharedMemorySize,
      static_cast<int>(shared_bytes)));
  CUDA_CHECK(cudaFuncSetCacheConfig(gemm_n4096_k14336_kernel, cudaFuncCachePreferShared));

  // Launch kernel
  gemm_n4096_k14336_kernel<<<grid, block, shared_bytes, stream>>>(A, B, C, M, lda, ldb, ldc);

  // Error check
  CUDA_CHECK(cudaGetLastError());
}

#include "kernel.h"
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <stdexcept>
#include <string>

// Validate tensor properties
static void validate_inputs(const torch::Tensor& A, const torch::Tensor& B) {
  if (!A.device().is_cuda() || !B.device().is_cuda()) {
    throw std::invalid_argument("Inputs A and B must be CUDA tensors.");
  }
  if (A.dtype() != torch::kHalf || B.dtype() != torch::kHalf) {
    throw std::invalid_argument("Inputs A and B must be torch.float16 (Half) tensors.");
  }
  if (A.dim() != 2 || B.dim() != 2) {
    throw std::invalid_argument("Inputs A and B must be 2D tensors.");
  }
  if (A.size(1) != GEMM_K) {
    throw std::invalid_argument("A must have shape [M, 14336].");
  }
  if (B.size(0) != GEMM_N || B.size(1) != GEMM_K) {
    throw std::invalid_argument("B must have shape [4096, 14336].");
  }
  if (A.device().index() != B.device().index()) {
    throw std::invalid_argument("A and B must be on the same CUDA device.");
  }
}

// Torch entry point
torch::Tensor run(torch::Tensor A, torch::Tensor B) {
  try {
    validate_inputs(A, B);

    // Ensure contiguous for coalesced access
    torch::Tensor A_contig = A.contiguous();
    torch::Tensor B_contig = B.contiguous();

    const int64_t M64 = A_contig.size(0);
    if (M64 <= 0) {
      throw std::invalid_argument("M must be > 0.");
    }
    if (GEMM_N != 4096 || GEMM_K != 14336) {
      throw std::runtime_error("Internal constants mismatch.");
    }

    int M = static_cast<int>(M64);
    int lda = GEMM_K;
    int ldb = GEMM_K;
    int ldc = GEMM_N;

    // Allocate output tensor C [M, 4096] on same device, kHalf
    auto options = A_contig.options().dtype(torch::kHalf);
    torch::Tensor C = torch::empty({M64, static_cast<int64_t>(GEMM_N)}, options);

    // Get raw pointers
    const __half* A_ptr = reinterpret_cast<const __half*>(A_contig.data_ptr<at::Half>());
    const __half* B_ptr = reinterpret_cast<const __half*>(B_contig.data_ptr<at::Half>());
    __half* C_ptr = reinterpret_cast<__half*>(C.data_ptr<at::Half>());

    // Launch kernel on current CUDA stream
    cudaStream_t stream = at::cuda::getCurrentCUDAStream().stream();
    gemm_n4096_k14336_cuda(A_ptr, B_ptr, C_ptr, M, lda, ldb, ldc, stream);

    // Optional synchronization/error checking
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
      throw std::runtime_error(std::string("CUDA kernel launch failed: ") + cudaGetErrorString(err));
    }

    return C;
  } catch (const std::exception& e) {
    throw; // Let PyTorch capture and propagate the exception
  }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run, "Kernel execution function");
}
