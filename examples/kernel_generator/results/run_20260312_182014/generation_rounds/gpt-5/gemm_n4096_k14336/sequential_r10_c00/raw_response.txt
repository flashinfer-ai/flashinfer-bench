#include "kernel.h"
#include <cstdio>
#include <cassert>
#include <mma.h>

using namespace nvcuda;

#ifndef CUDA_CHECK
#define CUDA_CHECK(err) do { \
  cudaError_t err_ = (err); \
  if (err_ != cudaSuccess) { \
    printf("CUDA Error %s at %s:%d\n", cudaGetErrorString(err_), __FILE__, __LINE__); \
  } \
} while (0)
#endif

// WMMA/Tensor Core tiling parameters tuned for Ampere (A800)
static constexpr int CTA_M = 128;  // Block tile size in M
static constexpr int CTA_N = 128;  // Block tile size in N
static constexpr int K_TILE = 16;  // WMMA K dimension (must be 16)

// Warp mapping
static constexpr int WARPS_PER_BLOCK = 8;     // 8 warps => 256 threads/block
static constexpr int WARPS_PER_BLOCK_ROW = 4; // warps along M
static constexpr int WARPS_PER_BLOCK_COL = 2; // warps along N
static constexpr int WARP_ROW_TILES = 2;      // per warp tiles along M (2 * 16 = 32 rows)
static constexpr int WARP_COL_TILES = 4;      // per warp tiles along N (4 * 16 = 64 cols)

static_assert(WARPS_PER_BLOCK_ROW * WARPS_PER_BLOCK_COL == WARPS_PER_BLOCK, "Warp grid mismatch.");
static_assert(WARPS_PER_BLOCK_ROW * WARP_ROW_TILES * 16 == CTA_M, "CTA_M mismatch.");
static_assert(WARPS_PER_BLOCK_COL * WARP_COL_TILES * 16 == CTA_N, "CTA_N mismatch.");
static_assert(GEMM_N % CTA_N == 0, "CTA_N must divide GEMM_N.");
static_assert(GEMM_K % K_TILE == 0, "K_TILE must divide GEMM_K.");

__host__ __device__ __forceinline__ int ceil_div_int(int a, int b) {
  return (a + b - 1) / b;
}

__launch_bounds__(WARPS_PER_BLOCK * 32, 2)
__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         int M, int lda, int ldb, int ldc) {
  // Shared memory layout:
  //   - As: [CTA_M x K_TILE] in row-major with ldm = K_TILE
  //   - Bs: [K_TILE x CTA_N] in column-major with ldm = K_TILE
  //   - Cs: [CTA_M x CTA_N] as float buffer with ldm = CTA_N
  extern __shared__ unsigned char smem_buf[];
  __half* As = reinterpret_cast<__half*>(smem_buf);
  __half* Bs = As + (CTA_M * K_TILE);
  // Ensure float alignment (As+Bs total bytes is (CTA_M*K_TILE + K_TILE*CTA_N) * sizeof(__half)
  // which is multiple of 4; safe for float alignment).
  float* Cs = reinterpret_cast<float*>(Bs + (K_TILE * CTA_N));

  const int block_m = blockIdx.y * CTA_M;
  const int block_n = blockIdx.x * CTA_N;

  // Warp and lane identification
  const int lane_id = threadIdx.x;         // 0..31
  const int warp_id = threadIdx.y;         // 0..(WARPS_PER_BLOCK-1)
  const int linear_tid = warp_id * blockDim.x + lane_id; // 0..255

  // Initialize accumulators for each warp's tiles
  wmma::fragment<wmma::accumulator, 16, 16, 16, float> c_frags[WARP_ROW_TILES][WARP_COL_TILES];
  #pragma unroll
  for (int i = 0; i < WARP_ROW_TILES; ++i) {
    #pragma unroll
    for (int j = 0; j < WARP_COL_TILES; ++j) {
      wmma::fill_fragment(c_frags[i][j], 0.0f);
    }
  }

  const int num_k_tiles = GEMM_K / K_TILE; // 14336 / 16 = 896

  // Loop over K dimension tiles (size K_TILE = 16)
  for (int kt = 0; kt < num_k_tiles; ++kt) {
    // Cooperative load A tile [CTA_M x K_TILE] into As (row-major, ldm = K_TILE)
    // Map 2048 elements across 256 threads = 8 elements per thread
    #pragma unroll
    for (int it = 0; it < (CTA_M * K_TILE + WARPS_PER_BLOCK * 32 - 1) / (WARPS_PER_BLOCK * 32); ++it) {
      int idx = linear_tid + it * (WARPS_PER_BLOCK * 32);
      if (idx < (CTA_M * K_TILE)) {
        int a_row = idx / K_TILE;       // 0..127
        int a_col = idx % K_TILE;       // 0..15
        int g_m = block_m + a_row;
        int g_k = kt * K_TILE + a_col;

        __half val = __float2half(0.0f);
        if (g_m < M) {
          val = A[g_m * lda + g_k];
        }
        As[a_row * K_TILE + a_col] = val;
      }
    }

    // Cooperative load B tile [K_TILE x CTA_N] into Bs (column-major, ldm = K_TILE)
    // Map 2048 elements across 256 threads = 8 elements per thread
    #pragma unroll
    for (int it = 0; it < (K_TILE * CTA_N + WARPS_PER_BLOCK * 32 - 1) / (WARPS_PER_BLOCK * 32); ++it) {
      int idx = linear_tid + it * (WARPS_PER_BLOCK * 32);
      if (idx < (K_TILE * CTA_N)) {
        int b_col = idx / K_TILE;       // 0..127 (N-local)
        int b_row = idx % K_TILE;       // 0..15  (K-local)
        int g_n = block_n + b_col;
        int g_k = kt * K_TILE + b_row;

        // B is [N, K] row-major: element is B[g_n, g_k]
        __half val = B[g_n * ldb + g_k];
        // Store into shared mem as column-major with ldm = K_TILE
        Bs[b_row + b_col * K_TILE] = val;
      }
    }

    __syncthreads();

    // Compute using WMMA on staged tiles
    const int warp_row = warp_id % WARPS_PER_BLOCK_ROW; // 0..3
    const int warp_col = warp_id / WARPS_PER_BLOCK_ROW; // 0..1

    // Iterate over each warp's micro-tiles
    #pragma unroll
    for (int i = 0; i < WARP_ROW_TILES; ++i) {
      int a_row_off = (warp_row * WARP_ROW_TILES + i) * 16; // starting row in As
      wmma::fragment<wmma::matrix_a, 16, 16, 16, __half, wmma::row_major> a_frag;
      // Load the A fragment for this tile row (same across all N tiles in this warp for current kt)
      const __half* a_tile_ptr = As + a_row_off * K_TILE;
      wmma::load_matrix_sync(a_frag, a_tile_ptr, K_TILE);

      #pragma unroll
      for (int j = 0; j < WARP_COL_TILES; ++j) {
        int b_col_off = (warp_col * WARP_COL_TILES + j) * 16; // starting col in Bs
        wmma::fragment<wmma::matrix_b, 16, 16, 16, __half, wmma::col_major> b_frag;
        const __half* b_tile_ptr = Bs + b_col_off * K_TILE;
        wmma::load_matrix_sync(b_frag, b_tile_ptr, K_TILE);

        // MMA accumulate
        wmma::mma_sync(c_frags[i][j], a_frag, b_frag, c_frags[i][j]);
      }
    }

    __syncthreads();
  }

  // Store accumulators to shared memory Cs [CTA_M x CTA_N] as floats (row-major, ldm = CTA_N)
  const int warp_row = warp_id % WARPS_PER_BLOCK_ROW;
  const int warp_col = warp_id / WARPS_PER_BLOCK_ROW;
  #pragma unroll
  for (int i = 0; i < WARP_ROW_TILES; ++i) {
    int c_row_off = (warp_row * WARP_ROW_TILES + i) * 16; // row index within CTA_M
    #pragma unroll
    for (int j = 0; j < WARP_COL_TILES; ++j) {
      int c_col_off = (warp_col * WARP_COL_TILES + j) * 16; // col index within CTA_N
      float* c_tile_ptr = Cs + c_row_off * CTA_N + c_col_off;
      wmma::store_matrix_sync(c_tile_ptr, c_frags[i][j], CTA_N, wmma::mem_row_major);
    }
  }

  __syncthreads();

  // Convert and write back to global C as half, guard M tail
  // Map 128*128 = 16384 elements across 256 threads = 64 elements per thread
  #pragma unroll
  for (int it = 0; it < (CTA_M * CTA_N + WARPS_PER_BLOCK * 32 - 1) / (WARPS_PER_BLOCK * 32); ++it) {
    int idx = linear_tid + it * (WARPS_PER_BLOCK * 32);
    if (idx < (CTA_M * CTA_N)) {
      int c_row = idx / CTA_N; // 0..127
      int c_col = idx % CTA_N; // 0..127
      int g_m = block_m + c_row;
      int g_n = block_n + c_col;
      if (g_m < M) {
        float val = Cs[c_row * CTA_N + c_col];
        C[g_m * ldc + g_n] = __float2half_rn(val);
      }
    }
  }
}

void gemm_n4096_k14336_cuda(const __half* A,
                            const __half* B,
                            __half* C,
                            int M,
                            int lda, int ldb, int ldc,
                            cudaStream_t stream) {
  // Grid and block configuration
  const int grid_x = GEMM_N / CTA_N;               // 4096 / 128 = 32
  const int grid_y = ceil_div_int(M, CTA_M);
  dim3 grid(grid_x, grid_y, 1);
  dim3 block(32, WARPS_PER_BLOCK, 1); // 32 threads per warp, WARPS_PER_BLOCK warps

  // Dynamic shared memory size: As + Bs (half) + Cs (float)
  size_t shared_bytes = sizeof(__half) * (CTA_M * K_TILE + K_TILE * CTA_N)
                      + sizeof(float) * (CTA_M * CTA_N);

  // Set cache preference (prefer shared memory)
  CUDA_CHECK(cudaFuncSetCacheConfig(gemm_n4096_k14336_kernel, cudaFuncCachePreferShared));

  // On some GPUs, need to opt-in larger dynamic shared memory per block
  CUDA_CHECK(cudaFuncSetAttribute(
      gemm_n4096_k14336_kernel,
      cudaFuncAttributeMaxDynamicSharedMemorySize,
      static_cast<int>(shared_bytes)));

  // Launch kernel
  gemm_n4096_k14336_kernel<<<grid, block, shared_bytes, stream>>>(A, B, C, M, lda, ldb, ldc);

  // Error check
  CUDA_CHECK(cudaGetLastError());
}

#include "kernel.h"
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <stdexcept>
#include <string>

static void validate_inputs(const torch::Tensor& A, const torch::Tensor& B) {
  if (A.dim() != 2 || B.dim() != 2) {
    throw std::invalid_argument("Inputs A and B must be 2D tensors.");
  }
  if (A.dtype() != torch::kHalf || B.dtype() != torch::kHalf) {
    throw std::invalid_argument("Inputs A and B must be torch.float16 (Half) tensors.");
  }
  if (A.size(1) != GEMM_K) {
    throw std::invalid_argument("A must have shape [M, 14336] with dtype float16.");
  }
  if (B.size(0) != GEMM_N || B.size(1) != GEMM_K) {
    throw std::invalid_argument("B must have shape [4096, 14336] with dtype float16.");
  }
}

torch::Tensor run(torch::Tensor A, torch::Tensor B) {
  try {
    // Validate shapes and dtypes
    validate_inputs(A, B);

    // Ensure CUDA availability
    if (!torch::cuda::is_available()) {
      throw std::runtime_error("CUDA is not available, cannot run the GPU kernel.");
    }

    // Determine device handling and move tensors if necessary
    bool A_is_cuda = A.device().is_cuda();
    bool B_is_cuda = B.device().is_cuda();

    int device_index = A_is_cuda ? A.device().index()
                     : (B_is_cuda ? B.device().index()
                                  : at::cuda::getCurrentCUDAStream().device_index());
    torch::Device device(torch::kCUDA, device_index);

    torch::Tensor A_dev = A_is_cuda ? A : A.to(device, torch::kHalf);
    torch::Tensor B_dev = B_is_cuda ? B : B.to(device, torch::kHalf);

    if (A_dev.device().index() != B_dev.device().index()) {
      B_dev = B_dev.to(A_dev.device(), torch::kHalf);
    }

    // Ensure contiguous memory for coalesced access
    torch::Tensor A_contig = A_dev.contiguous();
    torch::Tensor B_contig = B_dev.contiguous();

    const int64_t M64 = A_contig.size(0);
    if (M64 <= 0) {
      throw std::invalid_argument("M must be > 0.");
    }

    // Leading dimensions (row-major strides)
    int lda = static_cast<int>(A_contig.stride(0)); // == K for contiguous [M, K]
    int ldb = static_cast<int>(B_contig.stride(0)); // == K for contiguous [N, K]

    // Allocate output tensor C [M, 4096] on same device, kHalf
    auto options = A_contig.options().dtype(torch::kHalf).device(A_contig.device());
    torch::Tensor C_dev = torch::empty({M64, static_cast<int64_t>(GEMM_N)}, options);

    // Raw pointers
    const __half* A_ptr = reinterpret_cast<const __half*>(A_contig.data_ptr<at::Half>());
    const __half* B_ptr = reinterpret_cast<const __half*>(B_contig.data_ptr<at::Half>());
    __half* C_ptr = reinterpret_cast<__half*>(C_dev.data_ptr<at::Half>());

    // Use actual stride for C
    int ldc = static_cast<int>(C_dev.stride(0));

    // Launch kernel on current CUDA stream
    cudaStream_t stream = at::cuda::getCurrentCUDAStream().stream();
    gemm_n4096_k14336_cuda(A_ptr, B_ptr, C_ptr, static_cast<int>(M64), lda, ldb, ldc, stream);

    // Error check
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
      throw std::runtime_error(std::string("CUDA kernel launch failed: ") + cudaGetErrorString(err));
    }

    // If original inputs were on CPU, return result to CPU
    if (!A_is_cuda && !B_is_cuda) {
      return C_dev.to(torch::kCPU);
    }
    return C_dev;
  } catch (const std::exception& e) {
    throw std::runtime_error(std::string("gemm_n4096_k14336 run() failed: ") + e.what());
  }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run, "Kernel execution function",
        pybind11::arg("A"),
        pybind11::arg("B"));
}
