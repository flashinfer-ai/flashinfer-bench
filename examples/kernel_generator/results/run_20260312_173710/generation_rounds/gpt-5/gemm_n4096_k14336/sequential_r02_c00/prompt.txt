// Kernel implements: C[M, N_CONST] = A[M, K_CONST] * B[N_CONST, K_CONST]^T
// A and B are half-precision; accumulation in float for better numerical stability.
__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         const int M)
{
    // Threadblock tile sizes
    constexpr int TB_M = BM; // 64
    constexpr int TB_N = BN; // 128
    constexpr int TB_K = BK; // 64

    // Per-thread micro tile sizes (TM x TN)
    constexpr int TM = 4;
    constexpr int TN = 8;

    // Shared memory with padding to reduce bank conflicts
    __shared__ __half As[TB_M][TB_K + 1];
    __shared__ __half Bs[TB_N][TB_K + 1];

    // Compute block origin
    const int block_m0 = blockIdx.y * TB_M;
    const int block_n0 = blockIdx.x * TB_N;

    // Thread coordinates within block
    const int tx = threadIdx.x; // 0..15
    const int ty = threadIdx.y; // 0..15
    const int linear_tid = ty * blockDim.x + tx; // 0..255

    // Accumulators (float)
    float acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    // Loop over K dimension in tiles of TB_K
    for (int k0 = 0; k0 < K_CONST; k0 += TB_K) {
        // Load A tile [TB_M x TB_K] into shared memory
        {
            constexpr int elemsA = TB_M * TB_K; // 4096
            constexpr int loads_per_thread_A = elemsA / (blockDim.x * blockDim.y); // 16
            #pragma unroll
            for (int i = 0; i < loads_per_thread_A; ++i) {
                int idx = linear_tid + i * (blockDim.x * blockDim.y);
                int r = idx / TB_K; // 0..63
                int c = idx % TB_K; // 0..63
                int gr = block_m0 + r;
                int gc = k0 + c;
                __half val = __float2half(0.0f);
                if (gr < M) {
                    // A is [M x K_CONST] row-major
                    val = A[gr * K_CONST + gc];
                }
                As[r][c] = val;
            }
        }

        // Load B tile [TB_N x TB_K] into shared memory
        {
            constexpr int elemsB = TB_N * TB_K; // 8192
            constexpr int loads_per_thread_B = elemsB / (blockDim.x * blockDim.y); // 32
            #pragma unroll
            for (int i = 0; i < loads_per_thread_B; ++i) {
                int idx = linear_tid + i * (blockDim.x * blockDim.y);
                int r = idx / TB_K; // 0..127
                int c = idx % TB_K; // 0..63
                int gn = block_n0 + r; // 0..4095 (in-bounds by design)
                int gk = k0 + c;
                // B is [N_CONST x K_CONST] row-major
                Bs[r][c] = B[gn * K_CONST + gk];
            }
        }

        __syncthreads();

        // Compute on the loaded tiles
        #pragma unroll 8
        for (int kk = 0; kk < TB_K; ++kk) {
            // Load a fragment of A for this thread
            float a_frag[TM];
            #pragma unroll
            for (int im = 0; im < TM; ++im) {
                int ar = ty * TM + im;
                a_frag[im] = __half2float(As[ar][kk]);
            }

            // Load a fragment of B for this thread (TN columns)
            float b_frag[TN];
            #pragma unroll
            for (int jn = 0; jn < TN; ++jn) {
                int br = tx * TN + jn;
                b_frag[jn] = __half2float(Bs[br][kk]);
            }

            // FMA
            #pragma unroll
            for (int im = 0; im < TM; ++im) {
                float a_val = a_frag[im];
                #pragma unroll
                for (int jn = 0; jn < TN; ++jn) {
                    acc[im][jn] += a_val * b_frag[jn];
                }
            }
        }

        __syncthreads();
    }

    // Write results to C
    #pragma unroll
    for (int im = 0; im < TM; ++im) {
        int gr = block_m0 + ty * TM + im;
        if (gr < M) {
            #pragma unroll
            for (int jn = 0; jn < TN; ++jn) {
                int gc = block_n0 + tx * TN + jn; // always in [0, N_CONST)
                C[gr * N_CONST + gc] = __float2half_rn(acc[im][jn]);
            }
        }
    }
}
