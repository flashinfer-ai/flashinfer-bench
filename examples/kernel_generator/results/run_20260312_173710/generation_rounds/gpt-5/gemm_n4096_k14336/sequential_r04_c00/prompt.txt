constexpr int THREADS_PER_BLOCK = TB_THREADS_X * TB_THREADS_Y;

// Ensure perfect coverage for shared memory loads
static_assert((BM * BK) % THREADS_PER_BLOCK == 0, "A tile must be divisible by threads per block");
static_assert((BK * BN) % THREADS_PER_BLOCK == 0, "B tile must be divisible by threads per block");

// Loads per thread
constexpr int LOADS_PER_THREAD_A = (BM * BK) / THREADS_PER_BLOCK; // 4096/256 = 16
constexpr int LOADS_PER_THREAD_B = (BK * BN) / THREADS_PER_BLOCK; // 8192/256 = 32

// Kernel implements: C[M, N_CONST] = A[M, K_CONST] * B[N_CONST, K_CONST]^T
// A and B are half-precision; accumulation in float for numerical stability.
__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         const int M)
{
    // Threadblock tile sizes
    constexpr int TB_M = BM; // 64
    constexpr int TB_N = BN; // 128
    constexpr int TB_K = BK; // 64

    // Per-thread micro tile sizes (TM x TN)
    constexpr int TM = 4;
    constexpr int TN = 8;

    // Shared memory (float for better accumulation; fits exactly 48KB)
    __shared__ float As[TB_M][TB_K]; // 64*64*4 = 16KB
    __shared__ float Bs[TB_K][TB_N]; // 64*128*4 = 32KB

    // Compute block origin in output matrix C
    const int block_m0 = blockIdx.y * TB_M; // row start
    const int block_n0 = blockIdx.x * TB_N; // col start

    // Thread coordinates within block
    const int tx = threadIdx.x; // 0..15
    const int ty = threadIdx.y; // 0..15
    const int linear_tid = ty * blockDim.x + tx; // 0..255

    // Accumulators (float)
    float acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    // Loop over K dimension in tiles of TB_K
    for (int k0 = 0; k0 < K_CONST; k0 += TB_K) {
        // Load A tile [TB_M x TB_K] into shared memory (row-major)
        #pragma unroll
        for (int i = 0; i < LOADS_PER_THREAD_A; ++i) {
            int idx = linear_tid + i * THREADS_PER_BLOCK; // 0..4095
            int r = idx / TB_K; // 0..63
            int c = idx % TB_K; // 0..63
            int gr = block_m0 + r;
            int gk = k0 + c;
            float val = 0.0f;
            if (gr < M) {
                // A is [M x K_CONST] row-major
                val = __half2float(A[gr * K_CONST + gk]);
            }
            As[r][c] = val;
        }

        // Load B tile as transposed into shared memory as [TB_K x TB_N]:
        // Bs[kk][n] = B[n, k0 + kk]
        #pragma unroll
        for (int i = 0; i < LOADS_PER_THREAD_B; ++i) {
            int idx = linear_tid + i * THREADS_PER_BLOCK; // 0..8191
            int r = idx / TB_N; // 0..63 (kk within tile)
            int c = idx % TB_N; // 0..127 (n within tile)
            int gk = k0 + r;
            int gn = block_n0 + c; // valid by design
            Bs[r][c] = __half2float(B[gn * K_CONST + gk]); // B[n, k]
        }

        __syncthreads();

        // Compute on the loaded tiles
        // Each thread computes a TM x TN micro-tile
        const int ar0 = ty * TM;      // starting row within TB_M for this thread
        const int bn0 = tx * TN;      // starting col within TB_N for this thread

        #pragma unroll 8
        for (int kk = 0; kk < TB_K; ++kk) {
            float a_frag[TM];
            float b_frag[TN];

            // Load A fragment (TM values)
            #pragma unroll
            for (int im = 0; im < TM; ++im) {
                a_frag[im] = As[ar0 + im][kk];
            }

            // Load B fragment (TN values)
            #pragma unroll
            for (int jn = 0; jn < TN; ++jn) {
                b_frag[jn] = Bs[kk][bn0 + jn];
            }

            // FMA
            #pragma unroll
            for (int im = 0; im < TM; ++im) {
                float a_val = a_frag[im];
                #pragma unroll
                for (int jn = 0; jn < TN; ++jn) {
                    acc[im][jn] += a_val * b_frag[jn];
                }
            }
        }

        __syncthreads();
    }

    // Write results to C (convert float accumulators to half)
    const int ar0 = ty * TM;
    const int bn0 = tx * TN;
    #pragma unroll
    for (int im = 0; im < TM; ++im) {
        int gr = block_m0 + ar0 + im;
        if (gr < M) {
            #pragma unroll
            for (int jn = 0; jn < TN; ++jn) {
                int gc = block_n0 + bn0 + jn; // within [0, N_CONST)
                C[gr * N_CONST + gc] = __float2half_rn(acc[im][jn]);
            }
        }
    }
}
