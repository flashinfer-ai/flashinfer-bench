// Warp-level reduction for sum
static __inline__ __device__ float warp_sum(float val) {
    unsigned mask = 0xffffffffu;
    val += __shfl_down_sync(mask, val, 16);
    val += __shfl_down_sync(mask, val, 8);
    val += __shfl_down_sync(mask, val, 4);
    val += __shfl_down_sync(mask, val, 2);
    val += __shfl_down_sync(mask, val, 1);
    return val;
}

// Hidden size is fixed to 4096 by specification
constexpr int HIDDEN_SIZE = 4096;
constexpr float EPS = 1e-5f;

// Kernel: Each block processes one row (one batch element).
template <int BLOCK_SIZE>
__launch_bounds__(BLOCK_SIZE, 2)
__global__ void fused_add_rmsnorm_h4096_kernel(
    const __nv_bfloat16* __restrict__ hidden_states, // [B, 4096]
    const __nv_bfloat16* __restrict__ residual,      // [B, 4096]
    const __nv_bfloat16* __restrict__ weight,        // [4096]
    __nv_bfloat16* __restrict__ output,              // [B, 4096]
    int batch_size)
{
    int b = blockIdx.x;
    if (b >= batch_size) return;

    extern __shared__ float sdata[]; // dynamic shared memory
    float* s_x = sdata;                                // 4096 floats
    float* warp_sums = s_x + HIDDEN_SIZE;              // partial sums per warp

    int tid = threadIdx.x;
    int lane = tid & 31;
    int wid = tid >> 5; // warp id
    int num_warps = (BLOCK_SIZE + 31) / 32;

    // Base offset for this row
    int base = b * HIDDEN_SIZE;

    // 1) Load and fuse add into shared memory, accumulate sum of squares in FP32
    float local_sum = 0.0f;

    #pragma unroll
    for (int i = tid; i < HIDDEN_SIZE; i += BLOCK_SIZE) {
        __nv_bfloat16 h = hidden_states[base + i];
        __nv_bfloat16 r = residual[base + i];
        float x = __bfloat162float(h) + __bfloat162float(r);
        s_x[i] = x; // store for later reuse
        local_sum += x * x;
    }

    // 2) Block reduction to compute total sum of squares
    float sum = warp_sum(local_sum);
    if (lane == 0) {
        warp_sums[wid] = sum;
    }
    __syncthreads();

    if (wid == 0) {
        float v = (lane < num_warps) ? warp_sums[lane] : 0.0f;
        v = warp_sum(v);
        if (lane == 0) {
            warp_sums[0] = v;
        }
    }
    __syncthreads();
    float total_sum = warp_sums[0];

    // 3) Compute inverse RMS
    float mean = total_sum / static_cast<float>(HIDDEN_SIZE);
    float inv_rms = rsqrtf(mean + EPS);

    // 4) Normalize, multiply by weight, and write output as BF16
    #pragma unroll
    for (int i = tid; i < HIDDEN_SIZE; i += BLOCK_SIZE) {
        __nv_bfloat16 w_bf16 = weight[i];
        float w = __bfloat162float(w_bf16);
        float y = s_x[i] * inv_rms * w;
        output[base + i] = __float2bfloat16(y);
    }
}

// Launcher function
void fused_add_rmsnorm_h4096_launcher(
    const __nv_bfloat16* hidden_states,
    const __nv_bfloat16* residual,
    const __nv_bfloat16* weight,
    __nv_bfloat16* output,
    int batch_size,
    cudaStream_t stream)
{
    if (batch_size <= 0) {
        return;
    }

    // Choose a reasonable block size for A800 (Ampere) and memory footprint
    constexpr int BLOCK_SIZE = 256;

    // Shared memory size: 4096 floats (for x) + warp partial sums
    size_t smem_bytes = sizeof(float) * HIDDEN_SIZE
                      + sizeof(float) * ((BLOCK_SIZE + 31) / 32);

    dim3 grid(batch_size);
    dim3 block(BLOCK_SIZE);

    fused_add_rmsnorm_h4096_kernel<BLOCK_SIZE><<<grid, block, smem_bytes, stream>>>(
        hidden_states, residual, weight, output, batch_size
    );
    CUDA_CHECK(cudaGetLastError());
}

                  torch::Tensor residual,
                  torch::Tensor weight) {

    const int64_t batch_size = hs.size(0);

    }

    const __nv_bfloat16* rs_ptr = reinterpret_cast<const __nv_bfloat16*>(rs.data_ptr<at::BFloat16>());
    const __nv_bfloat16* wt_ptr = reinterpret_cast<const __nv_bfloat16*>(wt.data_ptr<at::BFloat16>());
    __nv_bfloat16* out_ptr = reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>());

    // Use current CUDA stream to integrate with PyTorch
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
