#define RMSNORM_H4096_KERNEL_H

#include <cuda_runtime.h>

#define HIDDEN_SIZE_4096 4096
#define RMSNORM_EPS 1e-5f

                                 const __nv_bfloat16* weight,
                                 __nv_bfloat16* output,

#ifndef CUDA_CHECK
#define CUDA_CHECK(call)                                                                 \
  do {                                                                                   \
    cudaError_t _e = (call);                                                             \
    if (_e != cudaSuccess) {                                                             \
      fprintf(stderr, "CUDA error %s at %s:%d: %s\n", #call, __FILE__, __LINE__,         \
              cudaGetErrorString(_e));                                                   \

// Warp-level reduction for sum using shuffle operations
__inline__ __device__ float warp_reduce_sum(float val) {
  unsigned mask = 0xffffffffu;
  val += __shfl_down_sync(mask, val, 16);
  val += __shfl_down_sync(mask, val, 8);
  val += __shfl_down_sync(mask, val, 4);
  val += __shfl_down_sync(mask, val, 2);
  val += __shfl_down_sync(mask, val, 1);
  return val;
}

// One block handles one row [hidden_size=4096]
// BLOCK_SIZE should be a multiple of 32 for warp-aligned execution.
template <int BLOCK_SIZE>
__launch_bounds__(BLOCK_SIZE, 2)
__global__ void rmsnorm_h4096_kernel(const __nv_bfloat16* __restrict__ hidden,
                                     const __nv_bfloat16* __restrict__ weight,
                                     __nv_bfloat16* __restrict__ output,
                                     int batch_size) {
  const int row = blockIdx.x;
  if (row >= batch_size) return;

  // Shared memory buffers:
  // - s_x: the row's hidden states converted to FP32 for reuse
  // - s_w: the weight vector converted to FP32 for fast reuse
  // - s_inv_rms: scalar inverse RMS for this row
  // - warp_sums: partial sums per warp
  __shared__ float s_x[HIDDEN_SIZE_4096];
  __shared__ float s_w[HIDDEN_SIZE_4096];
  __shared__ float s_inv_rms;
  __shared__ float warp_sums[32]; // supports up to 32 warps per block

  const int tid = threadIdx.x;
  const int lane = tid & 31;
  const int warp_id = tid >> 5;
  const int num_warps = (BLOCK_SIZE + 31) >> 5;

  const int base = row * HIDDEN_SIZE_4096;

  // Phase 1: load hidden (to FP32 in shared) and load weight (to FP32 in shared),
  //          compute partial sum of squares of hidden.
  float local_sum = 0.f;

  #pragma unroll 4
  for (int i = tid; i < HIDDEN_SIZE_4096; i += BLOCK_SIZE) {
    float xf = __bfloat162float(hidden[base + i]);
    s_x[i] = xf;
    local_sum += xf * xf;

    // Load weight once per block to shared memory, coalesced
    s_w[i] = __bfloat162float(weight[i]);
  }

  // Intra-warp reduction
  float warp_sum = warp_reduce_sum(local_sum);

  // Write per-warp partial sums
  if (lane == 0) {
    warp_sums[warp_id] = warp_sum;
  }
  __syncthreads();

  // First warp reduces across warps and computes inverse RMS
  if (warp_id == 0) {
    float v = (lane < num_warps) ? warp_sums[lane] : 0.f;
    float total = warp_reduce_sum(v);
    if (lane == 0) {
      float mean_sq = total / static_cast<float>(HIDDEN_SIZE_4096);
      s_inv_rms = rsqrtf(mean_sq + RMSNORM_EPS);
    }
  }
  __syncthreads();

  const float inv_rms = s_inv_rms;

  // Phase 2: apply normalization and weight, then store
  #pragma unroll 4
  for (int i = tid; i < HIDDEN_SIZE_4096; i += BLOCK_SIZE) {
    float y = (s_x[i] * inv_rms) * s_w[i];
    output[base + i] = __float2bfloat16_rn(y);
  }
}

void launch_rmsnorm_h4096_kernel(const __nv_bfloat16* hidden,
                                 const __nv_bfloat16* weight,
                                 __nv_bfloat16* output,
                                 int batch_size,
                                 cudaStream_t stream) {
  if (batch_size <= 0) return;

  // Tuned for A800 (SM80): balance occupancy and shared memory usage
  constexpr int BLOCK_SIZE = 512;
  dim3 grid(batch_size);
  dim3 block(BLOCK_SIZE);

  rmsnorm_h4096_kernel<BLOCK_SIZE><<<grid, block, 0, stream>>>(
      hidden, weight, output, batch_size);
  CUDA_CHECK(cudaGetLastError());
}

<cpp_file name="main.cpp">
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <stdexcept>
#include <string>
#include "kernel.h"

static void validate_inputs(const torch::Tensor& hidden_states,
                            const torch::Tensor& weight) {
  TORCH_CHECK(hidden_states.device().is_cuda(),
              "hidden_states must be a CUDA tensor");
  TORCH_CHECK(weight.device().is_cuda(),
              "weight must be a CUDA tensor");
  TORCH_CHECK(hidden_states.dtype() == torch::kBFloat16,
              "hidden_states must be dtype torch.bfloat16");
  TORCH_CHECK(weight.dtype() == torch::kBFloat16,
              "weight must be dtype torch.bfloat16");
  TORCH_CHECK(hidden_states.dim() == 2,
              "hidden_states must be 2D [batch_size, hidden_size]");
  TORCH_CHECK(weight.dim() == 1,
              "weight must be 1D [hidden_size]");
  TORCH_CHECK(hidden_states.size(1) == HIDDEN_SIZE_4096,
              "hidden_size must be exactly 4096, but got ", hidden_states.size(1));
  TORCH_CHECK(weight.size(0) == HIDDEN_SIZE_4096,
              "weight size must be exactly 4096, but got ", weight.size(0));
  TORCH_CHECK(hidden_states.device().index() == weight.device().index(),
              "hidden_states and weight must be on the same CUDA device");
}

torch::Tensor run(torch::Tensor hidden_states, torch::Tensor weight) {
  try {
    // Allow kwargs by accepting named arguments in binding (see PYBIND11_MODULE)
    validate_inputs(hidden_states, weight);

    // Guard the current device
    at::cuda::CUDAGuard device_guard(hidden_states.device());

    // Ensure contiguous memory for coalesced access
    auto x = hidden_states.contiguous();
    auto w = weight.contiguous();

    const int64_t batch_size = x.size(0);
    if (batch_size == 0) {
      return torch::empty_like(x);
    }

    // Allocate output tensor
    auto y = torch::empty_like(x);

    // Get current CUDA stream from PyTorch
    cudaStream_t stream = at::cuda::getCurrentCUDAStream().stream();

    // Raw pointers as __nv_bfloat16
    const __nv_bfloat16* x_ptr =
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>());
    const __nv_bfloat16* w_ptr =
        reinterpret_cast<const __nv_bfloat16*>(w.data_ptr<at::BFloat16>());
    __nv_bfloat16* y_ptr =
        reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>());

    // Launch kernel
    launch_rmsnorm_h4096_kernel(x_ptr, w_ptr, y_ptr, static_cast<int>(batch_size), stream);

    // Error check after launch
    cudaError_t err = cudaGetLastError();
    TORCH_CHECK(err == cudaSuccess,
                "CUDA kernel launch failed: ",
                cudaGetErrorString(err));

    return y;
  } catch (const c10::Error& e) {
    TORCH_CHECK(false, "ATen/Torch error in rmsnorm_h4096::run: ", e.msg());
  } catch (const std::exception& e) {
    TORCH_CHECK(false, "Exception in rmsnorm_h4096::run: ", e.what());
  } catch (...) {
    TORCH_CHECK(false, "Unknown error in rmsnorm_h4096::run");
  }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run,
        "Kernel execution function",
        py::arg("hidden_states"),
        py::arg("weight"));
}
