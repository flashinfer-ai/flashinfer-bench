// Warp-level reduction for sum using shuffle operations
__inline__ __device__ float warp_reduce_sum(float val) {
  unsigned mask = 0xffffffffu;
  // Unrolled for 32-wide warp
  val += __shfl_down_sync(mask, val, 16);
  val += __shfl_down_sync(mask, val, 8);
  val += __shfl_down_sync(mask, val, 4);
  val += __shfl_down_sync(mask, val, 2);
  val += __shfl_down_sync(mask, val, 1);
  return val;
}

// One block handles one row [hidden_size=4096]
template <int BLOCK_SIZE>
__launch_bounds__(BLOCK_SIZE, 2)
__global__ void rmsnorm_h4096_kernel(const __nv_bfloat16* __restrict__ hidden,
                                     __nv_bfloat16* __restrict__ output,
                                     int batch_size) {
  const int row = blockIdx.x;
  if (row >= batch_size) return;

  // Shared memory: store the input row to avoid reloading from global memory
  __shared__ __nv_bfloat16 s_x[HIDDEN_SIZE_4096];
  __shared__ float s_inv_rms;
  __shared__ float warp_sums[32]; // up to 32 warps per block

  const int tid = threadIdx.x;
  const int lane = tid & 31;
  const int warp_id = tid >> 5;
  const int num_warps = (BLOCK_SIZE + 31) >> 5;

  const int base = row * HIDDEN_SIZE_4096;

  // Phase 1: load, store to shared, and accumulate sum of squares in FP32
  float local_sum = 0.f;

  // Loop over hidden dimension with stride of BLOCK_SIZE
  for (int i = tid; i < HIDDEN_SIZE_4096; i += BLOCK_SIZE) {
    __nv_bfloat16 v = hidden[base + i];
    s_x[i] = v;
    float xf = __bfloat162float(v);
    local_sum += xf * xf;
  }

  // Reduce within warp
  float warp_sum = warp_reduce_sum(local_sum);

  // Write per-warp partial sums to shared memory
  if (lane == 0) {
    warp_sums[warp_id] = warp_sum;
  }
  __syncthreads();

  // First warp reduces the warp sums
  float total_sum = 0.f;
  if (warp_id == 0) {
    float wval = (lane < num_warps) ? warp_sums[lane] : 0.f;
    total_sum = warp_reduce_sum(wval);
    if (lane == 0) {
      float mean_sq = total_sum / static_cast<float>(HIDDEN_SIZE_4096);
      s_inv_rms = rsqrtf(mean_sq + RMSNORM_EPS);
    }
  }
  __syncthreads();

  const float inv_rms = s_inv_rms;

  // Phase 2: compute y = (x * inv_rms) * weight
  for (int i = tid; i < HIDDEN_SIZE_4096; i += BLOCK_SIZE) {
    float x = __bfloat162float(s_x[i]);
    float w = __bfloat162float(kWeight4096[i]);
    float y = (x * inv_rms) * w;
    output[base + i] = __float2bfloat16_rn(y);
  }
}

void launch_rmsnorm_h4096_kernel(const __nv_bfloat16* hidden,
                                 __nv_bfloat16* output,
                                 int batch_size,
                                 cudaStream_t stream) {
  if (batch_size <= 0) return;

  // Tune for A800/A100: 256 threads per block gives good occupancy and low register pressure
  constexpr int BLOCK_SIZE = 256;
  dim3 grid(batch_size);
  dim3 block(BLOCK_SIZE);

  rmsnorm_h4096_kernel<BLOCK_SIZE><<<grid, block, 0, stream>>>(hidden, output, batch_size);

  // Check launch and runtime errors
  CUDA_CHECK(cudaGetLastError());
}
