// Warp-level sum reduction using shuffles
__device__ __forceinline__ float warp_reduce_sum(float val) {
    unsigned mask = 0xffffffffu;
#pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(mask, val, offset);
    }
    return val;
}

// One block per row (batch element), 256 threads per block.
// Cache the row as float32 in shared memory to avoid reloading for the second pass.
__launch_bounds__(256, 4)
__global__ void rmsnorm_h4096_kernel(const __nv_bfloat16* __restrict__ hidden_states,
                                     const __nv_bfloat16* __restrict__ weight,
                                     __nv_bfloat16* __restrict__ output,
                                     int batch_size) {
    extern __shared__ float s_x[]; // dynamic shared memory sized to RMSNORM_HIDDEN_SIZE floats
    // We launch with 256 threads => 8 warps
    __shared__ float warp_sums[8];
    __shared__ float s_inv_rms;

    const int tid = threadIdx.x;
    const int row = blockIdx.x;

    if (row >= batch_size) return;

    constexpr int H = RMSNORM_HIDDEN_SIZE;
    constexpr int THREADS = 256;
    static_assert(H % THREADS == 0, "RMSNORM_HIDDEN_SIZE must be divisible by 256");
    static_assert(THREADS % 32 == 0, "THREADS must be a multiple of warp size");
    constexpr int ITERS = H / THREADS;

    const int base = row * H;

    // First pass: load BF16 -> FP32 into shared memory and accumulate sum of squares
    float sum = 0.0f;
#pragma unroll
    for (int i = 0; i < ITERS; ++i) {
        int idx = i * THREADS + tid; // 0..H-1
        float xi = __bfloat162float(hidden_states[base + idx]);
        s_x[idx] = xi;
        sum = fmaf(xi, xi, sum); // sum += xi * xi
    }
    __syncthreads();

    // Warp-level reduction
    sum = warp_reduce_sum(sum);

    const int warp_id = threadIdx.x >> 5; // /32
    const int lane_id = threadIdx.x & 31;

    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();

    // First warp reduces warp sums and computes inverse RMS
    if (warp_id == 0) {
        float total = (lane_id < (THREADS >> 5)) ? warp_sums[lane_id] : 0.0f;
        total = warp_reduce_sum(total);
        if (lane_id == 0) {
            constexpr float invH = 1.0f / static_cast<float>(H);
            float mean_sq = total * invH;
            s_inv_rms = rsqrtf(mean_sq + RMSNORM_EPS);
        }
    }
    __syncthreads();

    const float inv_rms = s_inv_rms;

    // Second pass: normalize with inv_rms and scale by weight
#pragma unroll
    for (int i = 0; i < ITERS; ++i) {
        int idx = i * THREADS + tid;
        float xi = s_x[idx];
        float wi = __bfloat162float(weight[idx]); // weight shared across rows
        float yi = xi * inv_rms * wi;
        output[base + idx] = __float2bfloat16_rn(yi);
    }
}
