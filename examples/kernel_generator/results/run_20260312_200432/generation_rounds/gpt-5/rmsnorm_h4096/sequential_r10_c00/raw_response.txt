<cuda_file name="kernel.cu">
#include "kernel.h"
#include <cuda.h>
#include <cuda_bf16.h>

// Store the per-layer weight in constant memory for fast, cached access across rows.
// Size is fixed (4096 bfloat16 = 8KB), fits comfortably in constant cache.
__constant__ __nv_bfloat16 c_weight[RMSNORM_HIDDEN_SIZE];

// Warp-level sum reduction using shuffles
__device__ __forceinline__ float warp_reduce_sum(float val) {
    unsigned mask = 0xffffffffu;
#pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(mask, val, offset);
    }
    return val;
}

// One block per row (batch element), 256 threads per block.
// Cache the row as float32 in shared memory to avoid reloading for the second pass.
// Weight is read from constant memory to reduce DRAM bandwidth for large batches.
__launch_bounds__(256, 4)
__global__ void rmsnorm_h4096_kernel(const __nv_bfloat16* __restrict__ hidden_states,
                                     const __nv_bfloat16* __restrict__ /*weight*/, // unused (constant memory used)
                                     __nv_bfloat16* __restrict__ output,
                                     int batch_size) {
    // Static shared memory for the row (4096 float32 = 16KB)
    __shared__ float s_x[RMSNORM_HIDDEN_SIZE];
    // Shared memory for warp partial sums and inverse RMS
    __shared__ float warp_sums[256 / 32];
    __shared__ float s_inv_rms;

    const int tid = threadIdx.x;
    const int row = blockIdx.x;

    if (row >= batch_size) return;

    constexpr int H = RMSNORM_HIDDEN_SIZE;
    constexpr int THREADS = 256;
    static_assert(H % THREADS == 0, "RMSNORM_HIDDEN_SIZE must be divisible by 256");
    static_assert(THREADS % 32 == 0, "THREADS must be a multiple of warp size");
    constexpr int ITERS = H / THREADS; // 16

    const int base = row * H;

    // First pass: load BF16 -> FP32 into shared memory and accumulate sum of squares
    float sum = 0.0f;
#pragma unroll
    for (int i = 0; i < ITERS; ++i) {
        int idx = i * THREADS + tid; // 0..H-1
        float xi = __bfloat162float(hidden_states[base + idx]);
        s_x[idx] = xi;
        sum = fmaf(xi, xi, sum); // sum += xi * xi
    }
    __syncthreads();

    // Warp-level reduction
    sum = warp_reduce_sum(sum);

    const int warp_id = threadIdx.x >> 5; // /32
    const int lane_id = threadIdx.x & 31;

    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();

    // First warp reduces warp sums and computes inverse RMS
    if (warp_id == 0) {
        float total = (lane_id < (THREADS >> 5)) ? warp_sums[lane_id] : 0.0f;
        total = warp_reduce_sum(total);
        if (lane_id == 0) {
            constexpr float invH = 1.0f / static_cast<float>(H);
            float mean_sq = total * invH;
            s_inv_rms = rsqrtf(mean_sq + RMSNORM_EPS);
        }
    }
    __syncthreads();

    const float inv_rms = s_inv_rms;

    // Second pass: normalize with inv_rms and scale by weight from constant memory
#pragma unroll
    for (int i = 0; i < ITERS; ++i) {
        int idx = i * THREADS + tid;
        float xi = s_x[idx];
        float wi = __bfloat162float(c_weight[idx]); // weight shared across rows
        float yi = xi * inv_rms * wi;
        output[base + idx] = __float2bfloat16_rn(yi);
    }
}

<cpp_file name="main.cpp">
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <stdexcept>
#include <sstream>
#include <limits>
#include <string>

#include "kernel.h"

// CUDA error checking helper
static inline void checkCudaOrThrow(const char* where) {
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        std::ostringstream oss;
        oss << where << " CUDA Error: " << cudaGetErrorString(err);
        throw std::runtime_error(oss.str());
    }
}

// Forward declaration of constant memory symbol from kernel.cu
extern __constant__ __nv_bfloat16 c_weight[];

// Upload weight tensor (BF16) to constant memory on the current device
static inline void upload_weight_to_constant(const __nv_bfloat16* w_ptr,
                                             size_t count_bytes,
                                             cudaStream_t stream) {
    cudaError_t st = cudaMemcpyToSymbolAsync(c_weight, w_ptr, count_bytes, 0,
                                             cudaMemcpyDeviceToDevice, stream);
    if (st != cudaSuccess) {
        std::ostringstream oss;
        oss << "cudaMemcpyToSymbolAsync(c_weight) failed: " << cudaGetErrorString(st);
        throw std::runtime_error(oss.str());
    }
}

torch::Tensor run(torch::Tensor hidden_states, torch::Tensor weight) {
    try {
        // Validate dtypes
        if (hidden_states.dtype() != torch::kBFloat16) {
            throw std::invalid_argument("hidden_states must have dtype torch.bfloat16.");
        }
        if (weight.dtype() != torch::kBFloat16) {
            throw std::invalid_argument("weight must have dtype torch.bfloat16.");
        }

        // Validate shapes
        if (hidden_states.dim() != 2) {
            throw std::invalid_argument("hidden_states must be a 2D tensor of shape [batch_size, 4096].");
        }
        if (weight.dim() != 1) {
            throw std::invalid_argument("weight must be a 1D tensor of shape [4096].");
        }

        const int64_t batch_size64 = hidden_states.size(0);
        const int64_t hidden_size = hidden_states.size(1);
        if (hidden_size != RMSNORM_HIDDEN_SIZE) {
            throw std::invalid_argument("hidden_size must be 4096.");
        }
        if (weight.size(0) != RMSNORM_HIDDEN_SIZE) {
            throw std::invalid_argument("weight must have size 4096.");
        }
        if (batch_size64 < 0 || batch_size64 > static_cast<int64_t>(std::numeric_limits<int>::max())) {
            throw std::invalid_argument("batch_size out of supported range for this kernel.");
        }
        const int batch_size = static_cast<int>(batch_size64);

        // Determine device index: use CUDA device if any tensor is CUDA, else current CUDA device
        int device_index = 0;
        if (hidden_states.device().is_cuda()) {
            device_index = hidden_states.get_device();
        } else if (weight.device().is_cuda()) {
            device_index = weight.get_device();
        } else {
            cudaGetDevice(&device_index);
        }
        c10::cuda::CUDAGuard device_guard(device_index);

        // Place tensors on the same CUDA device
        torch::Tensor hs_gpu = hidden_states;
        torch::Tensor w_gpu = weight;
        bool input_was_cpu = false;

        if (!hidden_states.device().is_cuda() || hidden_states.get_device() != device_index) {
            hs_gpu = hidden_states.to(torch::Device(torch::kCUDA, device_index));
            input_was_cpu = !hidden_states.device().is_cuda();
        }
        if (!weight.device().is_cuda() || weight.get_device() != device_index) {
            w_gpu = weight.to(torch::Device(torch::kCUDA, device_index));
            input_was_cpu = input_was_cpu || !weight.device().is_cuda();
        }

        // Ensure contiguous tensors for efficient memory access
        auto hs_contig = hs_gpu.is_contiguous() ? hs_gpu : hs_gpu.contiguous();
        auto w_contig = w_gpu.is_contiguous() ? w_gpu : w_gpu.contiguous();

        // Prepare output tensor on GPU
        auto output_gpu = torch::empty_like(hs_contig, hs_contig.options());

        // Get current CUDA stream from PyTorch
        auto stream = at::cuda::getCurrentCUDAStream();

        // Extract raw pointers (reinterpret to __nv_bfloat16)
        const __nv_bfloat16* hs_ptr = reinterpret_cast<const __nv_bfloat16*>(hs_contig.data_ptr<at::BFloat16>());
        const __nv_bfloat16* w_ptr  = reinterpret_cast<const __nv_bfloat16*>(w_contig.data_ptr<at::BFloat16>());
        __nv_bfloat16* out_ptr      = reinterpret_cast<__nv_bfloat16*>(output_gpu.data_ptr<at::BFloat16>());

        // Upload weight to constant memory for fast cached reads across blocks
        upload_weight_to_constant(w_ptr, RMSNORM_HIDDEN_SIZE * sizeof(__nv_bfloat16), stream.stream());

        // Configure kernel launch
        const dim3 threads(256);
        const dim3 blocks(static_cast<unsigned int>(batch_size));

        if (blocks.x > 0) {
            rmsnorm_h4096_kernel<<<blocks, threads, 0, stream.stream()>>>(
                hs_ptr, w_ptr, out_ptr, batch_size
            );
            checkCudaOrThrow("rmsnorm_h4096_kernel launch");
        }

        // If inputs were on CPU, synchronize and return CPU tensor to mirror input device
        if (input_was_cpu) {
            cudaError_t sync_status = cudaStreamSynchronize(stream.stream());
            if (sync_status != cudaSuccess) {
                std::ostringstream oss;
                oss << "Stream synchronization failed: " << cudaGetErrorString(sync_status);
                throw std::runtime_error(oss.str());
            }
            return output_gpu.to(torch::kCPU);
        }

        return output_gpu;
    } catch (const std::exception& e) {
        throw std::runtime_error(std::string("rmsnorm_h4096 run() failed: ") + e.what());
    }
}

// PyBind11 module binding
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("run", &run, "Kernel execution function");
}
