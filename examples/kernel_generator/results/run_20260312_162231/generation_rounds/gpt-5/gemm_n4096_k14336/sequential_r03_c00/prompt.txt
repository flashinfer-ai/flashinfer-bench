// Optimized GEMM for A800 (SM80+) using WMMA: C[M,N] = A[M,K] * B[N,K]^T
// A: row-major [M x K], B: row-major [N x K], C: row-major [M x N]
// Accumulation in FP32, inputs/outputs FP16
__global__ __launch_bounds__(gemm_n4096_k14336::THREADS_PER_BLOCK)
void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                              const __half* __restrict__ B,
                              __half* __restrict__ C,
                              int M) {
#if __CUDA_ARCH__ < 800
  return;
#endif
  // Compile-time constants
  constexpr int N = gemm_n4096_k14336::N_CONST;
  constexpr int K = gemm_n4096_k14336::K_CONST;

  constexpr int WMMA_M = gemm_n4096_k14336::WMMA_M;
  constexpr int WMMA_N = gemm_n4096_k14336::WMMA_N;
  constexpr int WMMA_K = gemm_n4096_k14336::WMMA_K;

  constexpr int TILE_M = gemm_n4096_k14336::TILE_M;
  constexpr int TILE_N = gemm_n4096_k14336::TILE_N;
  constexpr int TILE_K = gemm_n4096_k14336::TILE_K;

  // Threadblock coordinates
  const int block_row = blockIdx.y * TILE_M; // along M
  const int block_col = blockIdx.x * TILE_N; // along N

  // Shared memory panels
  __shared__ __align__(16) __half shmemA[TILE_M * TILE_K]; // [64 x 16], row-major, ld = 16
  __shared__ __align__(16) __half shmemB[TILE_K * TILE_N]; // [16 x 64], col-major, ld = 16

  // Each warp computes one 16x16 tile in the 64x64 output tile
  const int tid    = threadIdx.x;
  const int warpId = tid >> 5;       // 0..15
  const int laneId = tid & 31;

  constexpr int WARPS_M = TILE_M / WMMA_M; // 4
  constexpr int WARPS_N = TILE_N / WMMA_N; // 4
  const int warp_m = warpId / WARPS_N;     // 0..3
  const int warp_n = warpId % WARPS_N;     // 0..3

  // Initialize accumulator
  wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc_frag;
  wmma::fill_fragment(acc_frag, 0.0f);

  // Loop over K in steps of 16
  for (int k0 = 0; k0 < K; k0 += TILE_K) {
    // Cooperative load of A tile [TILE_M x TILE_K] into shared (row-major)
    for (int idx = tid; idx < TILE_M * TILE_K; idx += blockDim.x) {
      const int i = idx / TILE_K; // 0..63
      const int j = idx % TILE_K; // 0..15
      const int g_row = block_row + i;
      const int g_k   = k0 + j;
      __half val = __float2half(0.0f);
      if (g_row < M) {
        // A is [M x K], row-major
        val = A[g_row * K + g_k];
      }
      shmemA[i * TILE_K + j] = val;
    }

    // Cooperative load of B tile as B^T [TILE_K x TILE_N] into shared (col-major)
    // For (i,j) in [0..15]x[0..63], shmemB(i,j) = B[block_col + j, k0 + i]
    for (int idx = tid; idx < TILE_K * TILE_N; idx += blockDim.x) {
      const int i = idx % TILE_K; // 0..15
      const int j = idx / TILE_K; // 0..63
      const int g_n = block_col + j;
      const int g_k = k0 + i;
      // B is [N x K], row-major
      __half val = B[g_n * K + g_k];
      // Store as column-major with leading dimension TILE_K
      shmemB[i + j * TILE_K] = val;
    }

    __syncthreads();

    // Load fragments for this warp's 16x16 compute
    // A fragment (row-major) pointer: rows offset by warp_m*16, ld = TILE_K
    const __half* tileA = &shmemA[(warp_m * WMMA_M) * TILE_K];
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> a_frag;
    wmma::load_matrix_sync(a_frag, tileA, TILE_K);

    // B fragment (col-major) pointer: columns offset by warp_n*16, ld = TILE_K
    const __half* tileB = &shmemB[(warp_n * WMMA_N) * TILE_K];
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, __half, wmma::col_major> b_frag;
    wmma::load_matrix_sync(b_frag, tileB, TILE_K);

    // MMA accumulate
    wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);

    __syncthreads();
  }

  // Shared buffer to stage store safely for boundary on M
  __shared__ float shmemC[WARPS_PER_BLOCK * WMMA_M * WMMA_N]; // 16 * 256 floats
  float* warpC = &shmemC[warpId * WMMA_M * WMMA_N];
  wmma::store_matrix_sync(warpC, acc_frag, WMMA_N, wmma::mem_row_major);

  __syncthreads();

  // Convert to half and write to global C with row-boundary check
  const int tile_row_base = block_row + warp_m * WMMA_M;
  const int tile_col_base = block_col + warp_n * WMMA_N;

  for (int t = laneId; t < WMMA_M * WMMA_N; t += 32) {
    const int i = t / WMMA_N; // 0..15
    const int j = t % WMMA_N; // 0..15
    const int row = tile_row_base + i;
    const int col = tile_col_base + j;
    if (row < M) {
      const float valf = warpC[i * WMMA_N + j];
      C[row * N + col] = __float2half_rn(valf);
    }
  }
}
