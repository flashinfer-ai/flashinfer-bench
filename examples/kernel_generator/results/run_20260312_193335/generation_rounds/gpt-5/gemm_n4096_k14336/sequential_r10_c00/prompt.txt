// GEMM: C[M, N_CONST] = A[M, K_CONST] * (B[N_CONST, K_CONST])^T
// i.e., C[m, n] = sum_{k=0..K_CONST-1} A[m, k] * B[n, k]
// All tensors are in row-major layout.
// A: [M, K_CONST]
// B: [N_CONST, K_CONST]
// C: [M, N_CONST]

// Tiling parameters (correctness-first, tuned reasonably for Ampere/A800)
static constexpr int BM = 128; // M tile per block
static constexpr int BN = 128; // N tile per block
static constexpr int BK = 32;  // K tile per iteration (14336 % 32 == 0)

// Thread block: 16 x 16 = 256 threads
static constexpr int THREADS_X = 16; // along N
static constexpr int THREADS_Y = 16; // along M
static constexpr int THREADS_PER_BLOCK = THREADS_X * THREADS_Y;

// Each thread computes an 8 x 8 output sub-tile
static constexpr int TM = BM / THREADS_Y; // 8
static constexpr int TN = BN / THREADS_X; // 8

__device__ __forceinline__ float h2f(const __half h) {
    return __half2float(h);
}
__device__ __forceinline__ __half f2h(const float f) {
    return __float2half_rn(f);
}

__global__ void gemm_n4096_k14336_kernel(const __half* __restrict__ A,
                                         const __half* __restrict__ B,
                                         __half* __restrict__ C,
                                         int64_t M) {
    const int64_t block_m = static_cast<int64_t>(blockIdx.y) * BM;
    const int64_t block_n = static_cast<int64_t>(blockIdx.x) * BN;

    const int tx = threadIdx.x; // 0..15 (N)
    const int ty = threadIdx.y; // 0..15 (M)
    const int linear_tid = ty * blockDim.x + tx;

    const int row_start = ty * TM; // 0..127 step 8
    const int col_start = tx * TN; // 0..127 step 8

    // Shared memory:
    // As: [BM x BK] row-major
    // Bs_t: [BN x BK] row-major storing B transposed per tile: Bs_t[n_local, k_local] = B[g_n, g_k]
    extern __shared__ __half smem[];
    __half* As = smem;
    __half* Bs_t = As + (BM * BK);

    // Registers: FP32 accumulators
    float acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    // Iterate over K in BK-sized tiles
    for (int k0 = 0; k0 < K_CONST; k0 += BK) {
        // Cooperative load A tile [BM x BK]
        const int a_elems = BM * BK;
        for (int idx = linear_tid; idx < a_elems; idx += THREADS_PER_BLOCK) {
            int local_row = idx / BK;      // 0..BM-1
            int local_col = idx % BK;      // 0..BK-1
            int64_t g_m = block_m + local_row;
            int64_t g_k = k0 + local_col;
            __half v = __float2half_rn(0.0f);
            if (g_m < M) {
                v = A[g_m * K_CONST + g_k];
            }
            As[local_row * BK + local_col] = v;
        }

        // Cooperative load B tile as transposed into shared memory: [BN x BK]
        // Bs_t[local_n, local_k] = B[g_n, g_k]
        const int b_elems = BN * BK;
        for (int idx = linear_tid; idx < b_elems; idx += THREADS_PER_BLOCK) {
            int local_n = idx / BK;       // 0..BN-1
            int local_k = idx % BK;       // 0..BK-1
            int64_t g_n = block_n + local_n;
            int64_t g_k = k0 + local_k;
            __half v = __float2half_rn(0.0f);
            // N is exactly tiled to 4096, but guard anyway for robustness
            if (g_n < N_CONST) {
                v = B[g_n * K_CONST + g_k]; // B[n, k]
            }
            Bs_t[local_n * BK + local_k] = v;
        }

        __syncthreads();

        // Compute this K tile
        #pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            // Load fragments into registers
            float a_frag[TM];
            float b_frag[TN];

            // A fragment for TM rows corresponding to this thread
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                int ar = row_start + i; // 0..127
                a_frag[i] = h2f(As[ar * BK + kk]);
            }
            // B fragment for TN columns corresponding to this thread
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                int bn_local = col_start + j; // 0..127
                b_frag[j] = h2f(Bs_t[bn_local * BK + kk]);
            }

            // FMA accumulation
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                float a_val = a_frag[i];
                #pragma unroll
                for (int j = 0; j < TN; ++j) {
                    acc[i][j] = fmaf(a_val, b_frag[j], acc[i][j]);
                }
            }
        }

        __syncthreads();
    }

    // Store results
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        int64_t g_m = block_m + row_start + i;
        if (g_m >= M) continue;
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            int64_t g_n = block_n + col_start + j;
            // N is exactly tiled to 4096, but guard for robustness
            if (g_n < N_CONST) {
                C[g_m * N_CONST + g_n] = f2h(acc[i][j]);
            }
        }
    }
}

              static_cast<unsigned int>((M + BM - 1) / BM),

    size_t smem_bytes = (static_cast<size_t>(BM) * BK + static_cast<size_t>(BN) * BK) * sizeof(__half);
