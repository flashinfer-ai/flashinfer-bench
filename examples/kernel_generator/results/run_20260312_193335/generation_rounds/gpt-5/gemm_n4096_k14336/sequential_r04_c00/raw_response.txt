#include "kernel.h"
#include <cstdio>
#include <cstdint>

#ifndef CUDA_CHECK
#define CUDA_CHECK(stmt)                                                       \
    do {                                                                       \
        cudaError_t err = stmt;                                                \
        if (err != cudaSuccess) {                                              \
            fprintf(stderr, "CUDA error %s at %s:%d: %s\n", #stmt, __FILE__,   \
                    __LINE__, cudaGetErrorString(err));                        \
        }                                                                      \
    } while (0)
#endif

// Tiling parameters chosen for correctness and performance on A800.
// Compute C[M, 4096] = A[M, 14336] * (B[4096, 14336])^T
// Block-level tiles
static constexpr int BM = 128;  // Block tile in M
static constexpr int BN = 128;  // Block tile in N
static constexpr int BK = 64;   // Block tile in K (14336 / 64 = 224)

// Thread-level tiles: each thread computes an 8x8 tile of C within the 128x128 block
static constexpr int TM = 8;    // Thread tile M
static constexpr int TN = 8;    // Thread tile N

// Block threads: 16x16 = 256 threads
static constexpr int BLOCK_THREADS_X = 16;
static constexpr int BLOCK_THREADS_Y = 16;

__device__ __forceinline__ float h2f(const __half h) {
    return __half2float(h);
}

__device__ __forceinline__ __half f2h(const float f) {
    return __float2half_rn(f);
}

__global__ void gemm_n4096_k14336_tiled_kernel(const __half* __restrict__ A,
                                               const __half* __restrict__ B,
                                               __half* __restrict__ C,
                                               int64_t M) {
    // Determine block's starting coordinates in C
    const int64_t block_m = static_cast<int64_t>(blockIdx.y) * BM;
    const int64_t block_n = static_cast<int64_t>(blockIdx.x) * BN;

    // Thread coordinates and starting output sub-tile
    const int tx = threadIdx.x; // 0..15
    const int ty = threadIdx.y; // 0..15

    const int row_start = ty * TM; // within 0..120
    const int col_start = tx * TN; // within 0..120

    // Accumulators in FP32 for better numerical stability
    float acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    // Shared memory: A_sh stores [BM x BK] row-major
    //               B_sh stores [BK x BN] row-major (B_sh[k, n] is contiguous in n)
    extern __shared__ __half shmem[];
    __half* A_sh = shmem;
    __half* B_sh = A_sh + (BM * BK);

    // Flattened thread id for cooperative loading
    const int threads_per_block = blockDim.x * blockDim.y;
    const int linear_tid = ty * blockDim.x + tx;

    // Iterate over K dimension in tiles of BK
    for (int k0 = 0; k0 < K_CONST; k0 += BK) {
        // Load A tile [BM x BK] from A[block_m: block_m+BM, k0:k0+BK]
        const int a_elems = BM * BK;
        for (int idx = linear_tid; idx < a_elems; idx += threads_per_block) {
            int local_row = idx / BK;      // 0..BM-1
            int local_col = idx % BK;      // 0..BK-1
            int64_t g_m = block_m + local_row;
            int64_t g_k = k0 + local_col;
            __half val = __float2half(0.0f);
            if (g_m < M && g_k < K_CONST) {
                // A is [M, K_CONST], row-major
                val = A[g_m * K_CONST + g_k];
            }
            A_sh[local_row * BK + local_col] = val;
        }

        // Load B tile [BK x BN] from B (B is [N_CONST, K_CONST], row-major)
        // We need B^T in multiplication: C = A * B^T
        // So we load B_sh[k, n] = B[n, k]
        const int b_elems = BK * BN;
        for (int idx = linear_tid; idx < b_elems; idx += threads_per_block) {
            int local_k = idx / BN;       // 0..BK-1
            int local_n = idx % BN;       // 0..BN-1
            int64_t g_k = k0 + local_k;
            int64_t g_n = block_n + local_n;
            __half val = __float2half(0.0f);
            if (g_n < N_CONST && g_k < K_CONST) {
                // B is [N_CONST, K_CONST], row-major: B[g_n, g_k]
                val = B[g_n * K_CONST + g_k];
            }
            B_sh[local_k * BN + local_n] = val;
        }

        __syncthreads();

        // Compute the thread's 8x8 output sub-tile for this K tile
        #pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            float a_frag[TM];
            float b_frag[TN];

            // Load A fragment (TM elements from A_sh rows)
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                int ar = row_start + i;  // 0..127
                a_frag[i] = h2f(A_sh[ar * BK + kk]);
            }
            // Load B fragment (TN elements from B_sh columns)
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                int bn_local = col_start + j; // 0..127
                b_frag[j] = h2f(B_sh[kk * BN + bn_local]);
            }

            // FMA accumulate
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                float a_val = a_frag[i];
                #pragma unroll
                for (int j = 0; j < TN; ++j) {
                    acc[i][j] = fmaf(a_val, b_frag[j], acc[i][j]);
                }
            }
        }

        __syncthreads();
    }

    // Store results to global memory (convert to __half)
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        int64_t g_m = block_m + row_start + i;
        if (g_m >= M) continue; // guard M tail
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            int64_t g_n = block_n + col_start + j; // within [0, N_CONST)
            C[g_m * N_CONST + g_n] = f2h(acc[i][j]);
        }
    }
}

// Host launch function
void launch_gemm_n4096_k14336(const __half* A, const __half* B, __half* C, int64_t M, cudaStream_t stream) {
    // Configure grid and block
    dim3 block(BLOCK_THREADS_X, BLOCK_THREADS_Y, 1);
    dim3 grid((unsigned int)((N_CONST + BN - 1) / BN), (unsigned int)((M + BM - 1) / BM), 1);

    // Dynamic shared memory size: A_sh (BM*BK) + B_sh (BK*BN)
    size_t smem_bytes = (static_cast<size_t>(BM) * BK + static_cast<size_t>(BK) * BN) * sizeof(__half);

    // Launch kernel
    gemm_n4096_k14336_tiled_kernel<<<grid, block, smem_bytes, stream>>>(A, B, C, M);
    CUDA_CHECK(cudaGetLastError());
}

<cpp_file name="main.cpp">
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <iostream>
#include <stdexcept>
#include "kernel.h"

static void check_inputs(const torch::Tensor& A, const torch::Tensor& B) {
    // DType checks: FP16
    if (A.dtype() != torch::kFloat16 || B.dtype() != torch::kFloat16) {
        throw std::invalid_argument("Invalid dtype: A and B must be torch.float16 tensors.");
    }
    // Dim checks: 2D
    if (A.dim() != 2 || B.dim() != 2) {
        throw std::invalid_argument("Invalid shape: A and B must be 2D tensors.");
    }
    // Shape checks: A[M, K_CONST], B[N_CONST, K_CONST]
    if (A.size(1) != K_CONST) {
        throw std::invalid_argument("Invalid shape: A must have shape [M, 14336].");
    }
    if (B.size(0) != N_CONST || B.size(1) != K_CONST) {
        throw std::invalid_argument("Invalid shape: B must have shape [4096, 14336].");
    }
}

torch::Tensor run(torch::Tensor A_in, torch::Tensor B_in) {
    try {
        // Allow positional or keyword arguments; pybind supports named args automatically.
        check_inputs(A_in, B_in);

        bool inputs_on_cuda = A_in.device().is_cuda() && B_in.device().is_cuda();
        bool inputs_on_cpu = !A_in.device().is_cuda() && !B_in.device().is_cuda();

        if (!inputs_on_cuda && !inputs_on_cpu) {
            throw std::invalid_argument("Device mismatch: A and B must be on the same device (both CPU or both CUDA).");
        }

        // Ensure contiguous tensors
        torch::Tensor A_c = A_in.contiguous();
        torch::Tensor B_c = B_in.contiguous();

        int64_t M = A_c.size(0);
        if (M <= 0) {
            throw std::invalid_argument("Invalid shape: M must be > 0.");
        }

        // Prepare output tensor on the appropriate device (CUDA for kernel execution)
        torch::Tensor C_cuda;
        if (inputs_on_cuda) {
            // On GPU already: ensure same device
            if (A_c.get_device() != B_c.get_device()) {
                throw std::invalid_argument("Device mismatch: A and B must be on the same CUDA device.");
            }
            C_cuda = torch::empty({M, static_cast<int64_t>(N_CONST)}, A_c.options());
        } else {
            // Move to CUDA synchronously
            torch::Device cuda_dev(torch::kCUDA, at::cuda::current_device());
            A_c = A_c.to(cuda_dev, /*non_blocking=*/false);
            B_c = B_c.to(cuda_dev, /*non_blocking=*/false);
            C_cuda = torch::empty({M, static_cast<int64_t>(N_CONST)}, A_c.options());
        }

        // Raw pointers and stream
        const at::Half* A_ptr_at = A_c.data_ptr<at::Half>();
        const at::Half* B_ptr_at = B_c.data_ptr<at::Half>();
        at::Half* C_ptr_at = C_cuda.data_ptr<at::Half>();

        const __half* A_ptr = reinterpret_cast<const __half*>(A_ptr_at);
        const __half* B_ptr = reinterpret_cast<const __half*>(B_ptr_at);
        __half* C_ptr = reinterpret_cast<__half*>(C_ptr_at);

        cudaStream_t stream = at::cuda::getCurrentCUDAStream();

        // Launch kernel
        launch_gemm_n4096_k14336(A_ptr, B_ptr, C_ptr, M, stream);

        // Synchronize for safety and error checking
        cudaError_t errSync = cudaStreamSynchronize(stream);
        if (errSync != cudaSuccess) {
            throw std::runtime_error(std::string("CUDA kernel execution failed: ") + cudaGetErrorString(errSync));
        }

        // Return tensor on the same device as inputs
        if (inputs_on_cuda) {
            return C_cuda;
        } else {
            // Move result back to CPU
            return C_cuda.to(torch::kCPU, /*non_blocking=*/false);
        }
    } catch (const std::exception& e) {
        throw; // Re-throw to Python for visibility
    } catch (...) {
        throw std::runtime_error("Unknown error in run()");
    }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("run", &run, "Kernel execution function");
}
