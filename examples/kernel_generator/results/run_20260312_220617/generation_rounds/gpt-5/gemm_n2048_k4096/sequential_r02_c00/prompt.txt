#include <cuda_fp16.h>
#include <cstdio>

using namespace nvcuda;

namespace gemm_n2048_k4096 {

static __device__ __forceinline__ int div_up(int a, int b) {
  return (a + b - 1) / b;
}

#ifndef __CUDA_ARCH__
#define __CUDA_ARCH__ 800
#endif

// Kernel configuration tuned for A800 (SM80)
constexpr int BM = 128;   // block tile size in M
constexpr int BN = 128;   // block tile size in N
constexpr int BK = 16;    // tile size in K (WMMA k-dim)
constexpr int WARP_SIZE = 32;
constexpr int WARPS_PER_BLOCK = 8;  // 256 threads/block
constexpr int THREADS_PER_BLOCK = WARPS_PER_BLOCK * WARP_SIZE;

__global__ __launch_bounds__(THREADS_PER_BLOCK, 2)
void gemm_kernel_wmma_row_col(
    const __half* __restrict__ A,   // [M, K] row-major
    const __half* __restrict__ B,   // [N, K] row-major, but we treat as (K x N) col-major
    __half* __restrict__ C,         // [M, N] row-major
    int M)                          // variable M
{
    constexpr int N = GEMM_N;
    constexpr int K = GEMM_K;

    int block_m = blockIdx.y; // along M
    int block_n = blockIdx.x; // along N

    int tid = threadIdx.x;
    int warp_id = tid / WARP_SIZE;
    int lane_id = tid & (WARP_SIZE - 1);

    // Warp tile mapping: 2 along M, 4 along N per block (2*4=8 warps)
    int warp_m = warp_id % 2;
    int warp_n = warp_id / 2; // 0..3

    extern __shared__ unsigned char smem_raw[];
    __half* smemA = reinterpret_cast<__half*>(smem_raw);                        // size: BM * BK (128*16)
    __half* smemB = smemA + (BM * BK);                                          // size: BK * BN (16*128), stored col-major
    float*  smemC = reinterpret_cast<float*>(smemB + (BK * BN));                // per-warp staging: WARPS_PER_BLOCK * 16*16 floats

    // Accumulators: each warp computes a 64x32 tile via 4x2 WMMA tiles of 16x16
    wmma::fragment<wmma::accumulator, 16, 16, 16, float> acc[4][2];
#pragma unroll
    for (int mi = 0; mi < 4; ++mi) {
#pragma unroll
        for (int nj = 0; nj < 2; ++nj) {
            wmma::fill_fragment(acc[mi][nj], 0.0f);
        }
    }

    int m_block_base = block_m * BM;
    int n_block_base = block_n * BN;

    // Loop over K in 16-wide chunks
    for (int k0 = 0; k0 < K; k0 += BK) {
        // Cooperative load A tile (BM x BK) into shared memory, row-major contiguous
        int a_elems = BM * BK; // 128 * 16 = 2048
        for (int idx = tid; idx < a_elems; idx += blockDim.x) {
            int row = idx / BK;    // [0, BM)
            int col = idx % BK;    // [0, BK)
            int m = m_block_base + row;
            __half val = __float2half(0.0f);
            if (m < M) {
                val = A[m * K + (k0 + col)];
            }
            smemA[row * BK + col] = val;
        }

        // Cooperative load B tile (BK x BN) into shared memory as column-major (so WMMA matrix_b col_major works)
        int b_elems = BK * BN; // 16 * 128 = 2048
        for (int idx = tid; idx < b_elems; idx += blockDim.x) {
            int colN = idx / BK;   // [0, BN)
            int rowK = idx % BK;   // [0, BK)
            int n = n_block_base + colN;
            // B is [N, K] row-major; element (n, k0+rowK)
            __half val = B[n * K + (k0 + rowK)];
            // Store in shared as col-major (K x N tile): index = col * ld + row
            smemB[colN * BK + rowK] = val;
        }

        __syncthreads();

        // Compute with WMMA for this K-slice
#pragma unroll
        for (int mi = 0; mi < 4; ++mi) {
            int a_row = warp_m * 64 + mi * 16; // within BM (0..127)
            const __half* a_tile_ptr = smemA + a_row * BK;

            wmma::fragment<wmma::matrix_a, 16, 16, 16, __half, wmma::row_major> a_frag;
            wmma::load_matrix_sync(a_frag, a_tile_ptr, BK);

#pragma unroll
            for (int nj = 0; nj < 2; ++nj) {
                int b_col = warp_n * 32 + nj * 16; // within BN (0..127)
                const __half* b_tile_ptr = smemB + b_col * BK;

                wmma::fragment<wmma::matrix_b, 16, 16, 16, __half, wmma::col_major> b_frag;
                wmma::load_matrix_sync(b_frag, b_tile_ptr, BK);

                wmma::mma_sync(acc[mi][nj], a_frag, b_frag, acc[mi][nj]);
            }
        }

        __syncthreads(); // reuse smemA/smemB for next K-tile
    }

    // Epilogue: store accumulators to C with per-warp staging in shared memory
#pragma unroll
    for (int mi = 0; mi < 4; ++mi) {
#pragma unroll
        for (int nj = 0; nj < 2; ++nj) {
            int c_row_base = m_block_base + warp_m * 64 + mi * 16;
            int c_col_base = n_block_base + warp_n * 32 + nj * 16;

            // Each warp uses its own 16x16 float tile in shared memory
            float* warp_smemC = smemC + warp_id * 16 * 16;
            wmma::store_matrix_sync(warp_smemC, acc[mi][nj], 16, wmma::mem_row_major);
            __syncwarp();

            // Convert to half and write to global memory
            for (int e = lane_id; e < 256; e += WARP_SIZE) {
                int r = e / 16;
                int c = e % 16;
                int gm = c_row_base + r;
                int gn = c_col_base + c; // gn < N always for valid tiles
                if (gm < M) {
                    __half hv = __float2half_rn(warp_smemC[e]);
                    C[gm * N + gn] = hv;
                }
            }
            __syncwarp();
        }
    }
}

void gemm_n2048_k4096_cuda(const at::Half* A_h, const at::Half* B_h, at::Half* C_h, int64_t M, cudaStream_t stream) {
    const __half* A = reinterpret_cast<const __half*>(A_h);
    const __half* B = reinterpret_cast<const __half*>(B_h);
    __half* C = reinterpret_cast<__half*>(C_h);

    dim3 block(THREADS_PER_BLOCK);
    int grid_x = GEMM_N / BN;           // 2048 / 128 = 16
    int grid_y = (int) ((M + BM - 1) / BM);
    dim3 grid(grid_x, grid_y);

    size_t smem_bytes = sizeof(__half) * (BM * BK + BK * BN) + sizeof(float) * (WARPS_PER_BLOCK * 16 * 16);

    gemm_kernel_wmma_row_col<<<grid, block, smem_bytes, stream>>>(A, B, C, static_cast<int>(M));
    CUDA_CHECK(cudaGetLastError());
}

torch::Tensor run(torch::Tensor A_in, torch::Tensor B_in) {

        validate_inputs(A_in, B_in);
    } catch (const std::exception& e) {
        throw;
    }

    // Determine device and move tensors to CUDA if needed
    torch::Device device = A_in.device();
    if (!device.is_cuda()) {
        // If A is on CPU, move both to CUDA
        device = torch::Device(torch::kCUDA, 0);
    }
    if (B_in.device().is_cuda() && A_in.device().is_cuda()) {
        if (A_in.device().index() != B_in.device().index()) {
            throw std::invalid_argument("A and B must be on the same CUDA device.");
        }
        device = A_in.device();
    }

    // Use CUDA guard
    c10::cuda::CUDAGuard device_guard(device);

    // Ensure contiguous and on device
    torch::Tensor A = A_in.to(device).contiguous();
    torch::Tensor B = B_in.to(device).contiguous();

    int64_t M = A.size(0);
    TORCH_CHECK(B.size(0) == GEMM_N, "B.size(0) must be 2048.");
    TORCH_CHECK(A.size(1) == GEMM_K && B.size(1) == GEMM_K, "K dimension must be 4096.");

    // Allocate output [M, N]
    auto options = torch::TensorOptions().dtype(torch::kFloat16).device(device);
    torch::Tensor C = torch::empty({M, (int64_t)GEMM_N}, options);

    // Launch kernel
    auto stream = at::cuda::getCurrentCUDAStream();
    try {
        gemm_n2048_k4096_cuda(
            reinterpret_cast<const at::Half*>(A.data_ptr<at::Half>()),
            reinterpret_cast<const at::Half*>(B.data_ptr<at::Half>()),
            reinterpret_cast<at::Half*>(C.data_ptr<at::Half>()),
            M,
            stream.stream());
    } catch (const std::exception& e) {
        throw;
    }

    // Synchronize and check errors
    CUDA_CHECK(cudaGetLastError());
    CUDA_CHECK(cudaStreamSynchronize(stream.stream()));
    return C;
}
