// Warp reduce helper (sum)
static __device__ __forceinline__ float warp_reduce_sum(float val) {
#pragma unroll
  for (int offset = 16; offset > 0; offset >>= 1) {
    val += __shfl_down_sync(0xffffffff, val, offset);
  }
  return val;
}

template<int HEAD_DIM>
static __device__ __forceinline__ void load_q_vector_4floats_per_thread(
    const __nv_bfloat16* __restrict__ q_ptr, int lane, float qreg[HEAD_DIM / 32]) {
  static_assert(HEAD_DIM % 32 == 0, "HEAD_DIM must be multiple of warp size");
#pragma unroll
  for (int i = 0; i < HEAD_DIM / 32; ++i) {
    int d = lane + i * 32;
    qreg[i] = __bfloat162float(q_ptr[d]);
  }
}

template<int HEAD_DIM>
static __device__ __forceinline__ float dot_qk_4floats_per_thread(
    const float qreg[HEAD_DIM / 32],
    const __nv_bfloat16* __restrict__ sK_row, // pointer to sK[k * HEAD_DIM]
    int lane) {
  float sum = 0.f;
#pragma unroll
  for (int i = 0; i < HEAD_DIM / 32; ++i) {
    int d = lane + i * 32;
    float kf = __bfloat162float(sK_row[d]);
    sum += qreg[i] * kf;
  }
  // Reduce across warp and broadcast to all lanes
  sum = warp_reduce_sum(sum);
  sum = __shfl_sync(0xffffffff, sum, 0);
  return sum;
}

void gqa_ragged_prefill_causal_kernel(

  constexpr int NUM_KV_HEADS = GQA_KV_HEADS;

        float exp_mt = __expf(m_tile - m_new);

        for (int i = 0; i < HEAD_DIM / 32; ++i) {
          acc_global[i] += acc_tile[i] * exp_mt;
        }

        m_global = m_new;

    } // tiles loop

    }

}

  if (q.size(1) != GQA_QO_HEADS) throw std::invalid_argument("q.shape[1] must be 32");
  if (q.size(2) != GQA_HEAD_DIM) throw std::invalid_argument("q.shape[2] must be 128");
}

static inline void check_sizes_kv(const torch::Tensor& kv, const char* name) {
  if (kv.dim() != 3) {
    std::ostringstream oss; oss << name << " must be 3D [total_kv, 8, 128]";
    throw std::invalid_argument(oss.str());
  }
  if (kv.size(1) != GQA_KV_HEADS) {
    std::ostringstream oss; oss << name << ".shape[1] must be 8";
    throw std::invalid_argument(oss.str());
  }
  if (kv.size(2) != GQA_HEAD_DIM) {
    std::ostringstream oss; oss << name << ".shape[2] must be 128";
    throw std::invalid_argument(oss.str());
  }
}

static inline void check_indptr(const torch::Tensor& indptr, const char* name) {
  if (indptr.dim() != 1) {
    std::ostringstream oss; oss << name << " must be 1D [len_indptr]";
    throw std::invalid_argument(oss.str());
  }
  if (indptr.dtype() != at::kInt) {
    std::ostringstream oss; oss << name << " must have dtype int32";
    throw std::invalid_argument(oss.str());
  }
}

static std::vector<torch::Tensor> run_impl(
    torch::Tensor q_in,
    torch::Tensor k_in,
    torch::Tensor v_in,
    torch::Tensor qo_indptr_in,
    torch::Tensor kv_indptr_in,
    c10::optional<float> sm_scale_opt_val) {

  // Dtype checks upfront (before device moves)
  check_dtype(q_in, at::kBFloat16, "q");
  check_dtype(k_in, at::kBFloat16, "k");
  check_dtype(v_in, at::kBFloat16, "v");
  check_indptr(qo_indptr_in, "qo_indptr");
  check_indptr(kv_indptr_in, "kv_indptr");

  // Ensure CUDA available; move to GPU if inputs are on CPU
  ensure_cuda_available();
  at::Device cuda_dev = at::Device(at::kCUDA, at::cuda::current_device());

  bool on_cuda = q_in.device().is_cuda();
  if (!on_cuda) {
    // Move all tensors to CUDA
    q_in = q_in.to(cuda_dev);
    k_in = k_in.to(cuda_dev);
    v_in = v_in.to(cuda_dev);
    qo_indptr_in = qo_indptr_in.to(cuda_dev);
    kv_indptr_in = kv_indptr_in.to(cuda_dev);
  }

  // Validate shapes and contiguity
  check_sizes_q(q_in);
  check_sizes_kv(k_in, "k");
  check_sizes_kv(v_in, "v");
  check_contiguous(q_in, "q");
  check_contiguous(k_in, "k");
  check_contiguous(v_in, "v");
  check_contiguous(qo_indptr_in, "qo_indptr");
  check_contiguous(kv_indptr_in, "kv_indptr");

  // sm_scale handling
  float sm_scale = 1.0f / std::sqrt((float)GQA_HEAD_DIM);
  if (sm_scale_opt_val.has_value()) {
    sm_scale = sm_scale_opt_val.value();
  }

  // Constraint checks: indptr[-1] equals totals
  const int64_t len_indptr = qo_indptr_in.size(0);
  if (len_indptr != kv_indptr_in.size(0)) {
    throw std::invalid_argument("qo_indptr and kv_indptr must have the same length.");
  }

  // Fetch last elements to CPU to verify constraints
  int32_t total_q_from_indptr = qo_indptr_in.index({len_indptr - 1}).to(at::kCPU).item<int32_t>();
  int32_t total_kv_from_indptr = kv_indptr_in.index({len_indptr - 1}).to(at::kCPU).item<int32_t>();
  const int64_t total_q = q_in.size(0);
  const int64_t total_kv = k_in.size(0);
  if (total_q_from_indptr != total_q) {
    std::ostringstream oss;
    oss << "Constraint violated: total_q (" << total_q << ") != qo_indptr[-1] ("
        << total_q_from_indptr << ")";
    throw std::invalid_argument(oss.str());
  }
  if (total_kv_from_indptr != total_kv) {
    std::ostringstream oss;
    oss << "Constraint violated: total_kv (" << total_kv << ") != kv_indptr[-1] ("
        << total_kv_from_indptr << ")";
    throw std::invalid_argument(oss.str());
  }

  // Allocate outputs on same device as q_in (CUDA now)
  auto options_bf16 = torch::TensorOptions().dtype(at::kBFloat16).device(q_in.device());
  auto options_f32  = torch::TensorOptions().dtype(at::kFloat).device(q_in.device());
  torch::Tensor output = torch::zeros({total_q, GQA_QO_HEADS, GQA_HEAD_DIM}, options_bf16);
  torch::Tensor lse = torch::full({total_q, GQA_QO_HEADS}, -INFINITY, options_f32);

  if (len_indptr > 1 && (total_q > 0) && (total_kv > 0)) {
    // Raw pointers
    const __nv_bfloat16* q_ptr  = reinterpret_cast<const __nv_bfloat16*>(q_in.data_ptr<at::BFloat16>());
    const __nv_bfloat16* k_ptr  = reinterpret_cast<const __nv_bfloat16*>(k_in.data_ptr<at::BFloat16>());
    const __nv_bfloat16* v_ptr  = reinterpret_cast<const __nv_bfloat16*>(v_in.data_ptr<at::BFloat16>());
    const int32_t* qo_indptr_ptr = qo_indptr_in.data_ptr<int32_t>();
    const int32_t* kv_indptr_ptr = kv_indptr_in.data_ptr<int32_t>();
    __nv_bfloat16* out_ptr = reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>());
    float* lse_ptr = lse.data_ptr<float>();

    // Launch kernel
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    launch_gqa_ragged_prefill_causal_h32_kv8_d128_kernel(
        q_ptr, k_ptr, v_ptr,
        qo_indptr_ptr, kv_indptr_ptr,
        static_cast<int>(len_indptr),
        sm_scale,
        out_ptr, lse_ptr,
        stream);
  }

  // Ensure errors are propagated
  cudaError_t err = cudaGetLastError();
  if (err != cudaSuccess) {
    std::ostringstream oss;
    oss << "CUDA kernel launch error: " << cudaGetErrorString(err);
    throw std::runtime_error(oss.str());
  }

  // If inputs were on CPU, move outputs back to CPU to match user expectation
  if (!on_cuda) {
    output = output.to(at::kCPU);
    lse = lse.to(at::kCPU);
  }

  return {output, lse};
}

std::vector<torch::Tensor> run(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v,
    torch::Tensor qo_indptr,
    torch::Tensor kv_indptr,
    c10::optional<torch::Tensor> sm_scale_opt) {

  c10::optional<float> val;
  if (sm_scale_opt.has_value()) {
    torch::Tensor smt = sm_scale_opt.value();
    if (smt.numel() != 1) {
      throw std::invalid_argument("sm_scale must be a scalar tensor with one element.");
    }
    if (smt.dtype() != at::kFloat) {
      throw std::invalid_argument("sm_scale must be float32 dtype.");
    }
    val = smt.to(at::kCPU).item<float>();
  }
  return run_impl(q, k, v, qo_indptr, kv_indptr, val);
}
