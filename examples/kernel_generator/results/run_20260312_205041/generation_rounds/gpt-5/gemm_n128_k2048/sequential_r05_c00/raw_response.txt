constexpr int GEMM_N = 128;
constexpr int GEMM_K = 2048;

// CUDA kernel launcher declaration

<cuda_file name="kernel.cu">
#include "kernel.h"

#include <mma.h>
#include <cstdio>
#include <stdexcept>

using namespace nvcuda;

#ifndef CUDA_CHECK
#define CUDA_CHECK(err) do { \
  cudaError_t err_ = (err); \
  if (err_ != cudaSuccess) { \
    fprintf(stderr, "CUDA Error %s at %s:%d: %s\n", #err, __FILE__, __LINE__, cudaGetErrorString(err_)); \
    throw std::runtime_error(cudaGetErrorString(err_)); \
  } \
} while (0)
#endif

// Utility for integer division rounding up
__host__ __device__ inline int ceil_div(int a, int b) {
  return (a + b - 1) / b;
}

// Kernel configuration tuned for A800 (SM80)
// We pick a balanced tile to keep high parallelism for small M and good TC utilization.
constexpr int WMMA_M = 16;
constexpr int WMMA_N = 16;
constexpr int WMMA_K = 16;

constexpr int BLOCK_M = 128; // 8 WMMA tiles along M
constexpr int BLOCK_N = 32;  // 2 WMMA tiles along N (GEMM_N=128 => 4 blocks along N)

static_assert(GEMM_K % WMMA_K == 0, "GEMM_K must be divisible by WMMA_K");
static_assert(BLOCK_M % WMMA_M == 0, "BLOCK_M must be divisible by WMMA_M");
static_assert(BLOCK_N % WMMA_N == 0, "BLOCK_N must be divisible by WMMA_N");
static_assert(GEMM_N % BLOCK_N == 0, "GEMM_N must be divisible by BLOCK_N");

// The actual CUDA kernel using WMMA on Tensor Cores
__global__ void gemm_n128_k2048_kernel(const half* __restrict__ A,
                                       const half* __restrict__ B,
                                       half* __restrict__ C,
                                       int M,
                                       int lda, // leading dimension of A (K = 2048)
                                       int ldb, // leading dimension of B when viewed as KxN col-major (K = 2048)
                                       int ldc  // leading dimension of C (N = 128)
                                       ) {
#if __CUDA_ARCH__ < 800
  // Guard: this kernel is for SM80+. Do nothing on older arch to avoid misbehavior.
  return;
#endif
  // Shared memory layout:
  // A_smem: [BLOCK_M * WMMA_K] half  = 128 * 16 = 2048 half  (4 KB)
  // B_smem: [WMMA_K * BLOCK_N] half  = 16  * 32 = 512  half  (1 KB)
  // C_smem: [WARPS * WMMA_M * WMMA_N] float = (8*2) * 16 * 16 = 4096 float (16 KB)
  extern __shared__ unsigned char smem_raw[];
  half* A_smem = reinterpret_cast<half*>(smem_raw);
  half* B_smem = A_smem + (BLOCK_M * WMMA_K);
  float* Cwarp_smem = reinterpret_cast<float*>(B_smem + (WMMA_K * BLOCK_N));

  const int tid = threadIdx.x;
  const int warp_id = tid >> 5;   // warp within block
  const int lane_id = tid & 31;   // lane within warp

  // Block coordinates
  const int m_block = blockIdx.x * BLOCK_M;
  const int n_block = blockIdx.y * BLOCK_N;

  // Tail handling in M
  int tile_rows = M - m_block;
  if (tile_rows < 0) tile_rows = 0;
  if (tile_rows > BLOCK_M) tile_rows = BLOCK_M;
  const bool full_m_tile = (tile_rows == BLOCK_M);

  // Determine warp tile coordinates in 2D (warp rows x warp cols)
  constexpr int WARPS_M = BLOCK_M / WMMA_M; // 8
  constexpr int WARPS_N = BLOCK_N / WMMA_N; // 2
  constexpr int WARPS_PER_BLOCK = WARPS_M * WARPS_N; // 16

  const int warp_m = warp_id % WARPS_M;  // 0..7
  const int warp_n = warp_id / WARPS_M;  // 0..1

  const int active_warp_rows = full_m_tile ? WARPS_M : ceil_div(tile_rows, WMMA_M);
  const bool warp_active = (warp_m < active_warp_rows);

  // Initialize accumulator
  wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc_frag;
  wmma::fill_fragment(acc_frag, 0.0f);

  // Main loop over K in steps of 16
  for (int k0 = 0; k0 < GEMM_K; k0 += WMMA_K) {
    // Stage A tile [tile_rows x 16] into shared memory (row-major, ldm=WMMA_K)
    // Use all threads in the block for coalesced loading.
    int a_elems = tile_rows * WMMA_K;
    for (int idx = tid; idx < a_elems; idx += blockDim.x) {
      int r = idx / WMMA_K;  // 0..tile_rows-1
      int c = idx % WMMA_K;  // 0..15
      A_smem[r * WMMA_K + c] = __ldg(A + (m_block + r) * lda + (k0 + c));
    }

    // Stage B tile [16 x BLOCK_N] into shared memory in column-major layout with ldm=WMMA_K.
    // Original B is [N, K] row-major; view as [K, N] col-major with leading dim K.
    for (int idx = tid; idx < WMMA_K * BLOCK_N; idx += blockDim.x) {
      int r = idx % WMMA_K;  // 0..15 (k within tile)
      int c = idx / WMMA_K;  // 0..BLOCK_N-1 (n within tile)
      B_smem[r + c * WMMA_K] = __ldg(B + (n_block + c) * ldb + (k0 + r));
    }

    __syncthreads();

    if (warp_active) {
      // Load A fragment for this warp from shared memory (row-major)
      const half* a_ptr = A_smem + (warp_m * WMMA_M) * WMMA_K;
      wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
      wmma::load_matrix_sync(a_frag, a_ptr, WMMA_K);

      // Load B fragment for this warp from shared memory (col-major)
      const half* b_ptr = B_smem + (warp_n * WMMA_N) * WMMA_K;
      wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::col_major> b_frag;
      wmma::load_matrix_sync(b_frag, b_ptr, WMMA_K);

      // MMA accumulate
      wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
    }

    __syncthreads(); // allow overwriting smem for next K tile
  } // end K loop

  // Store results: each active warp writes its 16x16 tile
  if (!warp_active) return;

  // Per-warp buffer for conversion (float -> half)
  float* warp_buf = Cwarp_smem + warp_id * (WMMA_M * WMMA_N);

  // Store accumulator to per-warp buffer (row-major, ldm=16)
  wmma::store_matrix_sync(warp_buf, acc_frag, WMMA_N, wmma::mem_row_major);
  __syncwarp();

  // Convert to half and write to global memory
  // Each warp has 32 lanes; 16x16=256 elements; 8 elements per lane
  for (int t = lane_id; t < (WMMA_M * WMMA_N); t += 32) {
    int r = t / WMMA_N; // 0..15
    int c = t % WMMA_N; // 0..15
    int global_m = m_block + warp_m * WMMA_M + r;
    int global_n = n_block + warp_n * WMMA_N + c;
    if (global_m < M) {
      float v = warp_buf[r * WMMA_N + c];
      C[global_m * ldc + global_n] = __float2half_rn(v);
    }
  }
}

// Kernel launcher
void gemm_n128_k2048_launcher(const half* A, const half* B, half* C, int M, cudaStream_t stream) {
  if (M <= 0) return;

  // 16 warps per block -> 512 threads
  dim3 block(32 * (BLOCK_M / WMMA_M) * (BLOCK_N / WMMA_N), 1, 1); // 32 * 8 * 2 = 512
  dim3 grid(ceil_div(M, BLOCK_M), GEMM_N / BLOCK_N, 1); // split along N for concurrency (4 blocks along N)

  int lda = GEMM_K;
  int ldb = GEMM_K; // viewing B as KxN col-major with leading dimension K
  int ldc = GEMM_N;

  // Dynamic shared memory size calculation
  size_t smem_a = BLOCK_M * WMMA_K * sizeof(half);                             // 128*16 half
  size_t smem_b = WMMA_K * BLOCK_N * sizeof(half);                             // 16*32 half
  size_t smem_cwarp = (BLOCK_M / WMMA_M) * (BLOCK_N / WMMA_N) * WMMA_M * WMMA_N * sizeof(float); // 8*2*16*16 float
  size_t smem_bytes = smem_a + smem_b + smem_cwarp;

  // Increase max dynamic shared memory size attribute if necessary (Ampere allows >48KB)
  CUDA_CHECK(cudaFuncSetAttribute(
      gemm_n128_k2048_kernel,
      cudaFuncAttributeMaxDynamicSharedMemorySize,
      static_cast<int>(smem_bytes)
  ));

  gemm_n128_k2048_kernel<<<grid, block, smem_bytes, stream>>>(A, B, C, M, lda, ldb, ldc);
  CUDA_CHECK(cudaGetLastError());
}

<cpp_file name="main.cpp">
#include "kernel.h"

#include <torch/extension.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <stdexcept>
#include <sstream>

static void check_inputs(const torch::Tensor& A, const torch::Tensor& B) {
  if (A.dim() != 2 || B.dim() != 2) {
    throw std::invalid_argument("A and B must be 2D tensors");
  }
  if (A.size(1) != GEMM_K) {
    std::ostringstream oss;
    oss << "A must have shape [M, " << GEMM_K << "], but got [" << A.size(0) << ", " << A.size(1) << "]";
    throw std::invalid_argument(oss.str());
  }
  if (B.size(0) != GEMM_N || B.size(1) != GEMM_K) {
    std::ostringstream oss;
    oss << "B must have shape [" << GEMM_N << ", " << GEMM_K << "], but got [" << B.size(0) << ", " << B.size(1) << "]";
    throw std::invalid_argument(oss.str());
  }
  if (A.dtype() != torch::kHalf || B.dtype() != torch::kHalf) {
    throw std::invalid_argument("A and B must be torch.float16 (Half) tensors");
  }
  if (A.device().type() != B.device().type()) {
    throw std::invalid_argument("A and B must be on the same device (both CPU or both CUDA)");
  }
}

torch::Tensor run(torch::Tensor A_in, torch::Tensor B_in) {
  try {
    check_inputs(A_in, B_in);

    bool input_on_cuda = A_in.device().is_cuda();
    torch::Tensor A_cuda = A_in;
    torch::Tensor B_cuda = B_in;

    if (!input_on_cuda) {
      A_cuda = A_in.to(torch::kCUDA, /*non_blocking=*/false);
      B_cuda = B_in.to(torch::kCUDA, /*non_blocking=*/false);
    }

    if (!A_cuda.is_contiguous()) A_cuda = A_cuda.contiguous();
    if (!B_cuda.is_contiguous()) B_cuda = B_cuda.contiguous();

    const int64_t M64 = A_cuda.size(0);
    if (M64 < 0) {
      throw std::invalid_argument("Invalid M dimension for A");
    }
    const int M = static_cast<int>(M64);

    auto options = torch::TensorOptions().dtype(torch::kHalf).device(A_cuda.device());
    torch::Tensor C_cuda = torch::empty({M64, static_cast<int64_t>(GEMM_N)}, options);

    // Raw pointers
    const at::Half* A_ptr = A_cuda.data_ptr<at::Half>();
    const at::Half* B_ptr = B_cuda.data_ptr<at::Half>();
    at::Half* C_ptr = C_cuda.data_ptr<at::Half>();

    // Current CUDA stream from PyTorch
    cudaStream_t stream = at::cuda::getCurrentCUDAStream().stream();

    // Launch kernel
    gemm_n128_k2048_launcher(
        reinterpret_cast<const half*>(A_ptr),
        reinterpret_cast<const half*>(B_ptr),
        reinterpret_cast<half*>(C_ptr),
        M,
        stream
    );

    // Check for launch errors
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
      throw std::runtime_error(std::string("CUDA kernel launch error: ") + cudaGetErrorString(err));
    }

    // If inputs were on CPU, move result back to CPU
    if (!input_on_cuda) {
      return C_cuda.to(torch::kCPU);
    }
    return C_cuda;
  } catch (const std::exception& e) {
    throw std::runtime_error(std::string("run() failed: ") + e.what());
  }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run, "Kernel execution function",
        pybind11::arg("A"),
        pybind11::arg("B"));
}
