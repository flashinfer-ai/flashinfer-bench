#include "kernel.h"

#include <mma.h>
#include <cstdio>
#include <stdexcept>
#include <stdint.h>

using namespace nvcuda;

#ifndef CUDA_CHECK
#define CUDA_CHECK(err) do { \
  cudaError_t err_ = (err); \
  if (err_ != cudaSuccess) { \
    fprintf(stderr, "CUDA Error %s at %s:%d: %s\n", #err, __FILE__, __LINE__, cudaGetErrorString(err_)); \
    throw std::runtime_error(cudaGetErrorString(err_)); \
  } \
} while (0)
#endif

// Utility for integer division rounding up
__host__ __device__ inline int ceil_div(int a, int b) {
  return (a + b - 1) / b;
}

// WMMA tile dims
constexpr int WMMA_M = 16;
constexpr int WMMA_N = 16;
constexpr int WMMA_K = 16;

// Vector width for global <-> shared copies: 8 half = 16B
constexpr int VEC_ELEMS = 8;

#if __CUDA_ARCH__ >= 800
__device__ __forceinline__ void cp_async_ca_shared_global_16(void* smem_ptr, const void* gmem_ptr) {
  uint32_t smem_addr = static_cast<uint32_t>(__cvta_generic_to_shared(smem_ptr));
  asm volatile(
      "cp.async.ca.shared.global [%0], [%1], %2;\n" :: "r"(smem_addr), "l"(gmem_ptr), "n"(16));
}
__device__ __forceinline__ void cp_async_commit() {
  asm volatile("cp.async.commit_group;\n" ::);
}
__device__ __forceinline__ void cp_async_wait_all() {
  asm volatile("cp.async.wait_group 0;\n" ::);
}
#endif

// Kernel computes C[M, N=128] = A[M, K=2048] * B[N=128, K=2048]^T
// Tiles: TILE_M x TILE_N, with WMMA 16x16x16 accumulations and double-buffered K-slices
template<int TILE_M, int TILE_N>
__global__ __launch_bounds__(1024, 1)
void gemm_kernel_wmma_doublebuf(const half* __restrict__ A,
                                const half* __restrict__ B,
                                half* __restrict__ C,
                                int M,
                                int lda, // leading dimension of A (K)
                                int ldb, // leading dimension of B (K)
                                int ldc  // leading dimension of C (N)
                                ) {
  static_assert(TILE_M % WMMA_M == 0, "TILE_M must be divisible by WMMA_M");
  static_assert(TILE_N % WMMA_N == 0, "TILE_N must be divisible by WMMA_N");
  static_assert(GEMM_K % WMMA_K == 0, "GEMM_K must be divisible by WMMA_K");
  static_assert(TILE_N <= GEMM_N, "TILE_N must be <= GEMM_N");

  constexpr int WARPS_M = TILE_M / WMMA_M;
  constexpr int WARPS_N = TILE_N / WMMA_N;
  constexpr int WARPS_PER_BLOCK = WARPS_M * WARPS_N;

  const int tid = threadIdx.x;
  const int warp_id = tid >> 5;
  const int lane_id = tid & 31;

  // Block tile coordinates
  const int m_block = blockIdx.x * TILE_M;
  const int n_block = blockIdx.y * TILE_N;

  // Number of valid rows in this M tile
  int tile_rows = M - m_block;
  if (tile_rows < 0) tile_rows = 0;
  if (tile_rows > TILE_M) tile_rows = TILE_M;
  const bool full_m_tile = (tile_rows == TILE_M);

  // Active warps on the M dimension
  const int active_warp_rows = full_m_tile ? WARPS_M : ceil_div(tile_rows, WMMA_M);
  const int warp_m = warp_id % WARPS_M;
  const int warp_n = warp_id / WARPS_M;
  const bool warp_active = (warp_m < active_warp_rows);

  // Shared memory layout:
  // [ A_smem[2]: 2 * (TILE_M x WMMA_K) half ] |
  // [ B_smem[2]: 2 * (WMMA_K x TILE_N) half ] |
  // [ Cwarp_smem: WARPS_PER_BLOCK * (WMMA_M x WMMA_N) floats ]
  extern __shared__ unsigned char smem_raw[];
  half* smem_half = reinterpret_cast<half*>(smem_raw);

  half* A_smem_stage0 = smem_half;
  half* A_smem_stage1 = A_smem_stage0 + (TILE_M * WMMA_K);
  half* B_smem_stage0 = A_smem_stage1 + (TILE_M * WMMA_K);
  half* B_smem_stage1 = B_smem_stage0 + (WMMA_K * TILE_N);
  float* Cwarp_smem = reinterpret_cast<float*>(B_smem_stage1 + (WMMA_K * TILE_N));

  // Accumulator fragment per warp (float)
  wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc_frag;
  wmma::fill_fragment(acc_frag, 0.0f);

  // Preload stage 0 (k0 = 0)
  int stage = 0;
  {
    // A tile [tile_rows x 16] segmented into 16B chunks (8 halves)
    const int vecs_per_row_A = WMMA_K / VEC_ELEMS; // 2
    const int total_vec_A = tile_rows * vecs_per_row_A;
    for (int idx = tid; idx < total_vec_A; idx += blockDim.x) {
      int r = idx / vecs_per_row_A;
      int v = idx % vecs_per_row_A;
      const void* gptr = reinterpret_cast<const void*>(
          A + (m_block + r) * lda + (0 + v * VEC_ELEMS));
      void* sptr = reinterpret_cast<void*>(
          A_smem_stage0 + r * WMMA_K + v * VEC_ELEMS);
      #if __CUDA_ARCH__ >= 800
        cp_async_ca_shared_global_16(sptr, gptr);
      #else
        // Fallback path: 16B copy
        *reinterpret_cast<uint4*>(sptr) = *reinterpret_cast<const uint4*>(gptr);
      #endif
    }
    // B tile [16 x TILE_N]
    const int vecs_per_col_B = WMMA_K / VEC_ELEMS; // 2
    const int total_vec_B = TILE_N * vecs_per_col_B;
    for (int idx = tid; idx < total_vec_B; idx += blockDim.x) {
      int n = idx / vecs_per_col_B;
      int v = idx % vecs_per_col_B;
      const void* gptr = reinterpret_cast<const void*>(
          B + (n_block + n) * ldb + (0 + v * VEC_ELEMS));
      void* sptr = reinterpret_cast<void*>(
          B_smem_stage0 + v * VEC_ELEMS + n * WMMA_K);
      #if __CUDA_ARCH__ >= 800
        cp_async_ca_shared_global_16(sptr, gptr);
      #else
        *reinterpret_cast<uint4*>(sptr) = *reinterpret_cast<const uint4*>(gptr);
      #endif
    }
    #if __CUDA_ARCH__ >= 800
      cp_async_commit();
      cp_async_wait_all();
    #endif
    __syncthreads();
  }

  // Main K loop over 16-wide slices
  for (int k0 = 0; k0 < GEMM_K; k0 += WMMA_K) {
    // Prepare next stage
    if (k0 + WMMA_K < GEMM_K) {
      // Toggle stage buffer
      int next_stage = stage ^ 1;

      // A next slice
      const int vecs_per_row_A = WMMA_K / VEC_ELEMS; // 2
      const int total_vec_A = tile_rows * vecs_per_row_A;
      for (int idx = tid; idx < total_vec_A; idx += blockDim.x) {
        int r = idx / vecs_per_row_A;
        int v = idx % vecs_per_row_A;
        const void* gptr = reinterpret_cast<const void*>(
            A + (m_block + r) * lda + (k0 + WMMA_K + v * VEC_ELEMS));
        half* A_stage_ptr = (next_stage == 0 ? A_smem_stage0 : A_smem_stage1);
        void* sptr = reinterpret_cast<void*>(
            A_stage_ptr + r * WMMA_K + v * VEC_ELEMS);
        #if __CUDA_ARCH__ >= 800
          cp_async_ca_shared_global_16(sptr, gptr);
        #else
          *reinterpret_cast<uint4*>(sptr) = *reinterpret_cast<const uint4*>(gptr);
        #endif
      }

      // B next slice
      const int vecs_per_col_B = WMMA_K / VEC_ELEMS; // 2
      const int total_vec_B = TILE_N * vecs_per_col_B;
      for (int idx = tid; idx < total_vec_B; idx += blockDim.x) {
        int n = idx / vecs_per_col_B;
        int v = idx % vecs_per_col_B;
        const void* gptr = reinterpret_cast<const void*>(
            B + (n_block + n) * ldb + (k0 + WMMA_K + v * VEC_ELEMS));
        half* B_stage_ptr = (next_stage == 0 ? B_smem_stage0 : B_smem_stage1);
        void* sptr = reinterpret_cast<void*>(
            B_stage_ptr + v * VEC_ELEMS + n * WMMA_K);
        #if __CUDA_ARCH__ >= 800
          cp_async_ca_shared_global_16(sptr, gptr);
        #else
          *reinterpret_cast<uint4*>(sptr) = *reinterpret_cast<const uint4*>(gptr);
        #endif
      }
      #if __CUDA_ARCH__ >= 800
        cp_async_commit();
      #endif
    }

    // Compute this K-slice using current stage
    if (warp_active) {
      const half* A_stage_ptr = (stage == 0 ? A_smem_stage0 : A_smem_stage1);
      const half* B_stage_ptr = (stage == 0 ? B_smem_stage0 : B_smem_stage1);

      // A fragment: row-major from A_smem, stride = WMMA_K
      const half* a_ptr = A_stage_ptr + (warp_m * WMMA_M) * WMMA_K;
      wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
      wmma::load_matrix_sync(a_frag, a_ptr, WMMA_K);

      // B fragment: col-major from B_smem, stride = WMMA_K
      const half* b_ptr = B_stage_ptr + (warp_n * WMMA_N) * WMMA_K;
      wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::col_major> b_frag;
      wmma::load_matrix_sync(b_frag, b_ptr, WMMA_K);

      // MMA accumulate
      wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
    }

    // Wait for next stage to be ready before switching buffers
    if (k0 + WMMA_K < GEMM_K) {
      #if __CUDA_ARCH__ >= 800
        cp_async_wait_all();
      #endif
      __syncthreads();
      stage ^= 1;
    }
  }

  // Store results: each warp writes its 16x16 tile
  if (!warp_active) return;

  // Each warp stores its 16x16 tile from acc_frag -> shared float buffer
  float* warp_buf = Cwarp_smem + warp_id * (WMMA_M * WMMA_N);
  wmma::store_matrix_sync(warp_buf, acc_frag, WMMA_N, wmma::mem_row_major);
  __syncwarp();

  // Convert to half and write to global (guard M boundary)
  for (int t = lane_id; t < (WMMA_M * WMMA_N); t += 32) {
    int r = t / WMMA_N;
    int c = t % WMMA_N;
    int global_m = m_block + warp_m * WMMA_M + r;
    int global_n = n_block + warp_n * WMMA_N + c;
    if (global_m < M) {
      float v = warp_buf[r * WMMA_N + c];
      C[global_m * ldc + global_n] = __float2half_rn(v);
    }
  }
}

// Kernel launcher
void gemm_n128_k2048_launcher(const half* A, const half* B, half* C, int M, cudaStream_t stream) {
  if (M <= 0) return;

  const int lda = GEMM_K;
  const int ldb = GEMM_K;
  const int ldc = GEMM_N;

  // Use TILE_N=128 to fully cover N and increase B reuse
  constexpr int TILE_N = 128;
  static_assert(GEMM_N % TILE_N == 0, "GEMM_N must be divisible by TILE_N");

  dim3 grid, block;
  size_t smem_bytes = 0;

  auto launch_variant = [&](int TILE_M){
    const int WARPS_M = TILE_M / WMMA_M;
    const int WARPS_N = TILE_N / WMMA_N;
    const int WARPS_PER_BLOCK = WARPS_M * WARPS_N;
    block = dim3(WARPS_PER_BLOCK * 32, 1, 1);
    grid = dim3(ceil_div(M, TILE_M), GEMM_N / TILE_N, 1);

    size_t smem_a = (size_t)2 * TILE_M * WMMA_K * sizeof(half);   // double-buffer A
    size_t smem_b = (size_t)2 * WMMA_K * TILE_N * sizeof(half);   // double-buffer B
    size_t smem_c = (size_t)WARPS_PER_BLOCK * WMMA_M * WMMA_N * sizeof(float);
    smem_bytes = smem_a + smem_b + smem_c;
  };

  // Choose TILE_M based on M to improve occupancy for small M
  if (M >= 64) {
    // TILE_M=64, TILE_N=128 => 32 warps (1024 threads)
    launch_variant(64);
    CUDA_CHECK(cudaFuncSetAttribute(
      gemm_kernel_wmma_doublebuf<64, TILE_N>,
      cudaFuncAttributeMaxDynamicSharedMemorySize,
      static_cast<int>(smem_bytes)
    ));
    gemm_kernel_wmma_doublebuf<64, TILE_N><<<grid, block, smem_bytes, stream>>>(
      A, B, C, M, lda, ldb, ldc
    );
  } else {
    // TILE_M=32, TILE_N=128 => 16 warps (512 threads) for small M
    launch_variant(32);
    CUDA_CHECK(cudaFuncSetAttribute(
      gemm_kernel_wmma_doublebuf<32, TILE_N>,
      cudaFuncAttributeMaxDynamicSharedMemorySize,
      static_cast<int>(smem_bytes)
    ));
    gemm_kernel_wmma_doublebuf<32, TILE_N><<<grid, block, smem_bytes, stream>>>(
      A, B, C, M, lda, ldb, ldc
    );
  }

  CUDA_CHECK(cudaGetLastError());
}

#include "kernel.h"

#include <torch/extension.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <stdexcept>
#include <sstream>

static void check_inputs(const torch::Tensor& A, const torch::Tensor& B) {
  if (A.dim() != 2 || B.dim() != 2) {
    throw std::invalid_argument("A and B must be 2D tensors");
  }
  if (A.size(1) != GEMM_K) {
    std::ostringstream oss;
    oss << "A must have shape [M, " << GEMM_K << "], but got [" << A.size(0) << ", " << A.size(1) << "]";
    throw std::invalid_argument(oss.str());
  }
  if (B.size(0) != GEMM_N || B.size(1) != GEMM_K) {
    std::ostringstream oss;
    oss << "B must have shape [" << GEMM_N << ", " << GEMM_K << "], but got [" << B.size(0) << ", " << B.size(1) << "]";
    throw std::invalid_argument(oss.str());
  }
  if (A.dtype() != torch::kHalf || B.dtype() != torch::kHalf) {
    throw std::invalid_argument("A and B must be torch.float16 (Half) tensors");
  }
  if (A.device().type() != B.device().type()) {
    throw std::invalid_argument("A and B must be on the same device (both CPU or both CUDA)");
  }
}

torch::Tensor run(torch::Tensor A_in, torch::Tensor B_in) {
  try {
    check_inputs(A_in, B_in);

    bool input_on_cuda = A_in.device().is_cuda();
    torch::Tensor A_cuda = A_in;
    torch::Tensor B_cuda = B_in;

    if (!input_on_cuda) {
      A_cuda = A_in.to(torch::kCUDA, /*non_blocking=*/false);
      B_cuda = B_in.to(torch::kCUDA, /*non_blocking=*/false);
    }

    if (!A_cuda.is_contiguous()) A_cuda = A_cuda.contiguous();
    if (!B_cuda.is_contiguous()) B_cuda = B_cuda.contiguous();

    const int64_t M64 = A_cuda.size(0);
    if (M64 < 0) {
      throw std::invalid_argument("Invalid M dimension for A");
    }
    const int M = static_cast<int>(M64);

    auto options = torch::TensorOptions().dtype(torch::kHalf).device(A_cuda.device());
    torch::Tensor C_cuda = torch::empty({M64, static_cast<int64_t>(GEMM_N)}, options);

    // Raw pointers
    const at::Half* A_ptr = A_cuda.data_ptr<at::Half>();
    const at::Half* B_ptr = B_cuda.data_ptr<at::Half>();
    at::Half* C_ptr = C_cuda.data_ptr<at::Half>();

    // Current CUDA stream from PyTorch
    cudaStream_t stream = at::cuda::getCurrentCUDAStream().stream();

    // Launch kernel
    gemm_n128_k2048_launcher(
        reinterpret_cast<const half*>(A_ptr),
        reinterpret_cast<const half*>(B_ptr),
        reinterpret_cast<half*>(C_ptr),
        M,
        stream
    );

    // Check for launch errors
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
      throw std::runtime_error(std::string("CUDA kernel launch error: ") + cudaGetErrorString(err));
    }

    // If inputs were on CPU, move result back to CPU
    if (!input_on_cuda) {
      return C_cuda.to(torch::kCPU);
    }
    return C_cuda;
  } catch (const std::exception& e) {
    throw std::runtime_error(std::string("run() failed: ") + e.what());
  }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("run", &run, "Kernel execution function",
        pybind11::arg("A"),
        pybind11::arg("B"));
}
